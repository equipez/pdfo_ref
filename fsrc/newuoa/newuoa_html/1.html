1     ! output_mod is a module providing some subroutines concerning output to
2     ! terminla/files. Note that these output operations are sequential in
3     ! nature. In case parallelisum is desirable (especially during
4     ! initializaton), the subroutines may have to be modified or disabled.

6     ! Coded by Zaikun Zhang in July 2020 based on Powell's Fortran 77 code 
7     ! and the NEWUOA paper.

19     subroutine retmssg(info, iprint, nf, f, x, solver)
20     use consts_mod, only : RP, IK, MSSGLEN, OUTUNIT
21     use info_mod, only : FTARGET_ACHIEVED, MAXFUN_REACHED
22     use info_mod, only : SMALL_TR_RADIUS, TRSUBP_FAILED
23     use info_mod, only : NAN_X, NAN_INF_F, NAN_MODEL
24     implicit none
25     
26     ! Inputs
27     integer(IK), intent(in) :: iprint 
28     integer(IK), intent(in) :: info
29     integer(IK), intent(in) :: nf
30     real(RP), intent(in) :: f
31     real(RP), intent(in) :: x(:) 
32     character(len = *), intent(in) :: solver
33     
34     ! Intermediate variables
35     integer :: ios  ! Should be an integer of default kind
36     character(len = 100) :: fout
37     character(len = 3) :: fstat
38     character(len = MSSGLEN) :: mssg
39     logical :: fexist
40     
41     
42     if (info == FTARGET_ACHIEVED) then
43         mssg = 'the target function value is achieved.'
44     else if (info == MAXFUN_REACHED) then
45         mssg = 'the objective function has been evaluated MAXFUN times.'
46     else if (info == SMALL_TR_RADIUS) then
47         mssg = 'the trust region radius reaches its lower bound.'
48     else if (info == TRSUBP_FAILED) then
49         mssg = 'a trust region step has failed to reduce the quadratic model.'
50     else if (info == NAN_X) then
51         mssg = 'NaN occurs in x.'
52     else if (info == NAN_INF_F) then
53         mssg = 'the objective function returns NaN or +INFINITY.' 
54     else if (info == NAN_MODEL) then
55         mssg = 'NaN occurs in the models.'
56     end if
57     
58     if (iprint >= 1) then
59         if (iprint >= 3) then
60             print '(1X)'
61         end if
62         print '(/4A)', 'Return from ', solver, ' because ', trim(mssg)
63         print '(1A, 6X, 1A, I7)', 'At the return from ' // solver, 'Number of function evaluations = ', nf
64         print '(1A, 1PD23.15, 6X, 1A, /(1P, 5D15.6))', 'Least function value = ', f, 'The corresponding X is:', x
65         print '(1X)'
66     end if
67     
68     if (iprint <= -1) then
69         fout = solver // '_output.txt'
70         inquire(file = trim(fout), exist = fexist)
71         if (fexist) then
72             fstat = 'old'
73         else
74             fstat = 'new'
75         end if
76         open(unit = OUTUNIT, file = trim(fout), status = fstat, position = 'append', iostat = ios, action = 'write')
77         if (ios /= 0) then
78             print '(1A)', 'Fail to open file ' // trim(fout) // '!'
79         else
80             if (iprint <= -3) then
81                 write(OUTUNIT, '(1X)')
82             end if
83             write(OUTUNIT, '(/4A)') 'Return from ', solver, ' because ', trim(mssg)
84             write(OUTUNIT, '(1A, 6X, 1A, I7)') 'At the return from ' // solver, 'Number of function evaluations = ', nf
85             write(OUTUNIT, '(1A, 1PD23.15, 6X, 1A, /(1P, 5D15.6))') 'Least function value = ', f, 'The corresponding X is:', x
86             close(OUTUNIT)     
87         end if
88         !print '(/1A /)', 'The output is printed to ' // trim(fout) // '.'
89     end if
90     
91     end subroutine retmssg

95     use consts_mod, only : RP, IK, OUTUNIT

112

114         if (iprint >= 3) then
115             print '(1X)'
116         end if

118         print '(1A, 1PD23.15, 6X, 1A, /(1P, 5D15.6))', 'Least function value = ', f, 'The corresponding X is:', x
119     end if
120     
121     if (iprint <= -2) then
122         fout = solver // '_output.txt'

124         if (fexist) then

126         else
127             fstat = 'new'
128         end if

130         if (ios /= 0) then
131             print '(1A)', 'Fail to open file ' // trim(fout) // '!'
132         else
133             if (iprint <= -3) then
134                 write(OUTUNIT, '(1X)')
135             end if
136             write(OUTUNIT, '(/1A, 1PD11.4, 6X, A, I7)') 'New RHO = ', rho, 'Number of function evaluations = ', nf
137             write(OUTUNIT, '(1A, 1PD23.15, 6X, 1A, /(1P, 5D15.6))') 'Least function value = ', f, 'The corresponding X is:', x
138             close(OUTUNIT)     
139         end if
140     end if
141     
142     end subroutine rhomssg
