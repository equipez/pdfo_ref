1     ! PINTRF_MOD is a module specifying the abstract interfaces FUNEVAL and 
2     ! FCEVAL. FUNEVAL evaluates the objective function for unconstrained, 
3     ! bound constrained, and linearly constrained problems; FCEVAL evaluates
4     ! the objective function and constraint for nonlinearly constrained prolems.
5     !
6     ! Coded by Zaikun Zhang in July 2020.
7     !
8     ! Users must provide the implementation of CALFUN or CALCFC.
9     
10     module pintrf_mod
11     
12     implicit none
13     private
14     public :: FUNEVAL, FCEVAL 
15     
16     abstract interface 
17         subroutine FUNEVAL(x, f)
18         use consts_mod, only : RP
19         implicit none
20         real(RP), intent(in) :: x(:)
21         real(RP), intent(out) :: f
22         end subroutine FUNEVAL
23     end interface
24     
25     
26     abstract interface
27         subroutine FCEVAL(x, f, con)
28         use consts_mod, only : RP
29         implicit none
30         real(RP), intent(in) :: x(:)
31         real(RP), intent(out) :: f
32         real(RP), intent(out) :: con(:)
33         end subroutine FCEVAL 
34     end interface
35     
36     end module pintrf_mod
