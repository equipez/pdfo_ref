1     ! VLAGBETA_MOD is a module providing a suroutine that calculates VLAG
2     ! and BETA for a given step D. 
3     ! Both VLAG and BETA are critical for the updating procedure of H, which
4     ! is detailed formula (4.11) of the NEWUOA paper. See (4.12) for the
5     ! definition of BETA, and VLAG is indeed Hw. 
6     !
7     ! Coded by Zaikun Zhang in July 2020 based on Powell's Fortran 77 code 
8     ! and the NEWUOA paper.
9     
10     
11     module vlagbeta_mod
12     
13     implicit none
14     private
15     public :: vlagbeta
16     
17     
18     contains
19     
20     
21     subroutine vlagbeta(idz, kopt, bmat, d, xopt, xpt, zmat, beta, vlag)
22     ! VLAGBETA is calculates VLAG = Hw and BETA for a given step D. 
23     ! See (4.11)--(4.12) of the NEWUOA paper.
24     
25     ! Generic modules
26     use consts_mod, only : RP, IK, ONE, HALF, DEBUGGING, SRNLEN
27     use debug_mod, only : errstop, verisize
28     use lina_mod, only : Ax_plus_y, xA_plus_y, xpy_dot_z, inprod, matprod
29     
30     implicit none
31     
32     ! Inputs
33     integer(IK), intent(in) :: idz
34     integer(IK), intent(in) :: kopt
35     real(RP), intent(in) :: bmat(:, :)  ! BMAT(N, NPT + N)
36     real(RP), intent(in) :: d(:)  ! D(N)
37     real(RP), intent(in) :: xopt(:)  ! XOPT(N)
38     real(RP), intent(in) :: xpt(:, :)  ! XPT(N, NPT)
39     real(RP), intent(in) :: zmat(:, :)  ! ZMAT(NPT, NPT - N - 1)
40     
41     ! Outputs
42     real(RP), intent(out) :: beta
43     real(RP), intent(out) :: vlag(:)  ! VLAG(NPT + N)
44     
45     ! Intermediate variables
46     integer(IK) :: n
47     integer(IK) :: npt
48     real(RP) :: bw(size(bmat, 1))
49     real(RP) :: bwvd
50     real(RP) :: dsq
51     real(RP) :: dx
52     real(RP) :: wcheck(size(zmat, 1)) 
53     real(RP) :: wz(size(zmat, 2))
54     real(RP) :: wzsave(size(wz))
55     real(RP) :: xoptsq
56     character(len = SRNLEN), parameter :: srname = 'VLAGBETA'
57     
58     
59     ! Get and verify the sizes
60     n = int(size(xpt, 1), kind(n))
61     npt = int(size(xpt, 2), kind(npt))
62     
63     if (DEBUGGING) then
64         if (n == 0 .or. npt < n + 2) then
65             call errstop(srname, 'SIZE(XPT) is invalid')
66         end if
67         call verisize(bmat, n, npt + n)
68         call verisize(zmat, npt, int(npt - n - 1, kind(n)))
69         call verisize(xopt, n)
70         call verisize(d, n)
71         call verisize(vlag, n + npt)
72     end if
73     
74     
75     !----------------------------------------------------------------------!
76     ! This is the one of the two places where WCHECK is calculated, 
77     ! the other one being BIGDEN (now removed). 
78     ! WCHECK contains the first NPT entries of (w-v) for the vectors 
79     ! w and v defined in eq(4.10) and eq(4.24) of the NEWUOA paper, 
80     ! and also hat{w} in eq(6.5) of 
81     !
82     ! M. J. D. Powell, Least Frobenius norm updating of quadratic
83     ! models that satisfy interpolation conditions. Math. Program., 
84     ! 100:183--215, 2004
85     wcheck = matprod(d, xpt)
86     wcheck = wcheck*(HALF*wcheck + matprod(xopt, xpt))
87     !----------------------------------------------------------------------!
88     
89     vlag(1 : npt) = matprod(d, bmat(:, 1 : npt))
90     
91     wz = matprod(wcheck, zmat)
92     wzsave = wz
93     wz(1 : idz - 1) = -wz(1 : idz - 1)
94     beta = -inprod(wzsave, wz)
95     !----------------------------------------------------------------------!
96     !-----!vlag(1 : npt) = vlag(1 : npt) + matprod(zmat, wz) !-------------!
97      vlag(1 : npt) = Ax_plus_y(zmat, wz, vlag(1 : npt))
98     !----------------------------------------------------------------------!
99     
100     bw = matprod(bmat(:, 1 : npt), wcheck)
101     !----------------------------------------------------------------------!
102     !vlag(npt + 1 : npt + n) = bw + matprod(d, bmat(:, npt + 1 : npt + n)) !
103     vlag(npt + 1: npt + n) = xA_plus_y(bmat(:, npt + 1 : npt + n), d, bw)
104     !----------------------------------------------------------------------!
105     
106     !----------------------------------------------------------------------!
107     !-----!bwvd = inprod(bw + vlag(npt + 1 : npt + n), d) !----------------!
108     bwvd = xpy_dot_z(bw, vlag(npt + 1 : npt + n), d)
109     !----------------------------------------------------------------------!
110     
111     dx = inprod(d, xopt)
112     
113     dsq = inprod(d, d)
114     xoptsq = inprod(xopt, xopt)
115     
116     beta = dx*dx + dsq*(xoptsq + dx + dx + HALF*dsq) + beta - bwvd
117     vlag(kopt) = vlag(kopt) + ONE
118     
119     end subroutine vlagbeta
120     
121     
122     end module vlagbeta_mod
