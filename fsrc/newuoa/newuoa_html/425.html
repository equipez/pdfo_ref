1     ! SHIFTBASE_MOD is a module contaning a subroutine that shifts the base
2     ! point from XBASE to XBASE + XPT.
3     !
4     ! Coded by Zaikun Zhang in July 2020 based on Powell's Fortran 77 code 
5     ! and the NEWUOA paper.
6     
7     
8     module shiftbase_mod
9     
10     implicit none
11     private
12     public :: shiftbase
13     
14     
15     contains 
16     
17     
18     subroutine shiftbase(idz, pq, zmat, bmat, gq, hq, xbase, xopt, xpt)
19     ! SHIFTBASE shifts the base point for XBASE to XBASE + XOPT and updates 
20     ! GQ, HQ, and BMAT accordingly. PQ and ZMAT remain the same after the
21     ! shifting. See Section 7 of the NEWUOA paper.
22     
23     ! Generic modules
24     use consts_mod, only : RP, IK, ZERO, ONE, HALF, QUART, DEBUGGING, SRNLEN
25     use debug_mod, only : errstop, verisize
26     use lina_mod, only : Ax_plus_y, r1update, r2update, inprod, matprod
27     
28     implicit none
29     
30     ! Inputs
31     integer(IK), intent(in) :: idz
32     real(RP), intent(in) :: pq(:)  ! PQ(NPT)
33     real(RP), intent(in) :: zmat(:, :)  ! ZMAT(NPT, NPT - N - 1)
34     
35     ! In-outputs
36     real(RP), intent(inout) :: bmat(:, :)  ! BMAT(N, NPT + N)
37     real(RP), intent(inout) :: gq(:)  ! GQ(N)
38     real(RP), intent(inout) :: hq(:, :)  ! HQ(N, N)
39     real(RP), intent(inout) :: xbase(:)  ! XBASE(N)
40     real(RP), intent(inout) :: xopt(:)  ! XOPT(N)
41     real(RP), intent(inout) :: xpt(:, :)  ! XPT(N, NPT)
42     
43     ! Intermediate variables
44     integer(IK) :: k
45     integer(IK) :: n
46     integer(IK) :: npt
47     real(RP) :: bmatk(size(bmat, 1)) 
48     real(RP) :: qxoptq
49     real(RP) :: sumz(size(zmat, 2))
50     real(RP) :: vlag(size(xopt))
51     real(RP) :: w1(size(pq))
52     real(RP) :: w2(size(xopt))
53     real(RP) :: xoptsq
54     real(RP) :: xpq(size(xopt))
55     character(len = SRNLEN), parameter :: srname = 'SHIFTBASE' 
56     
57     
58     ! Get and verify the sizes
59     n = int(size(xpt, 1), kind(n))
60     npt = int(size(xpt, 2), kind(npt))
61     
62     if (DEBUGGING) then
63         if (n == 0 .or. npt < n + 2) then
64             call errstop(srname, 'SIZE(XPT) is invalid')
65         end if
66         call verisize(pq, npt)
67         call verisize(zmat, npt, int(npt - n - 1, kind(n)))
68         call verisize(bmat, n, npt + n)
69         call verisize(gq, n)
70         call verisize(hq, n, n)
71         call verisize(xopt, n)
72         call verisize(xbase, n)
73     end if
74     
75           
76     xoptsq = inprod(xopt, xopt)
77     qxoptq = QUART * xoptsq
78     
79     !----------------------------------------------------------------!   
80     ! The update for gq can indeed be done by the following 3 lines:
81     !real(RP) :: hxopt(n)
82     !call hessmul(n, npt, xpt, hq, pq, xopt, hxopt)
83     !gq = gq + hxopt 
84     gq = Ax_plus_y(xpt, pq*matprod(xopt, xpt), gq)
85     gq = Ax_plus_y(hq, xopt, gq)
86     !----------------------------------------------------------------!
87     
88     w1 = matprod(xopt, xpt) - HALF*xoptsq
89     ! W1 equals MATPROD(XPT, XOPT) after XPT is updated as follows.
90     xpt = xpt - HALF*spread(xopt, dim = 2, ncopies = npt)
91     !do k = 1, npt
92     !    xpt(:, k) = xpt(:, k) - HALF*xopt
93     !end do
94     
95     ! Update HQ. It has to be done after the above revision to XPT!!!
96     xpq = matprod(xpt, pq)
97     
98     !----------------------------------------------------------------!
99     ! Implement R2UPDATE properly so that it ensures HQ is symmetric.
100     call r2update(hq, ONE, xopt, xpq)
101     !----------------------------------------------------------------!
102     
103     
104     ! Make the changes to BMAT that do not depend on ZMAT.
105     do k = 1, npt
106         bmatk = bmat(:, k)
107         w2 = w1(k)*xpt(:, k) + qxoptq*xopt
108         ! Implement R2UPDATE properly so that it ensures 
109         ! bmat(:, npt+1:npt+n) is symmetric.
110         call r2update(bmat(:, npt+1 : npt+n), ONE, bmatk, w2)
111     end do
112     
113     ! Then the revisions of BMAT that depend on ZMAT are calculated.
114     sumz = sum(zmat, dim = 1)
115     do k = 1, int(idz - 1, kind(k))
116     !----------------------------------------------------------------------!
117     !---------!vlag = qxoptq*sumz(k)*xopt + matprod(xpt, w1*zmat(:, k)) !--!
118         vlag = Ax_plus_y(xpt, w1*zmat(:, k), qxoptq*sumz(k)*xopt)
119     !----------------------------------------------------------------------!
120         call r1update(bmat(:, 1 : npt), -ONE, vlag, zmat(:, k))
121         ! Implement R1UPDATE properly so that it ensures 
122         ! bmat(:, npt+1:npt+n) is symmetric.
123         call r1update(bmat(:, npt + 1 : npt + n), -ONE, vlag)
124     end do
125     do k = idz, int(npt - n - 1, kind(k))
126     !----------------------------------------------------------------------!
127     !---------!vlag = qxoptq*sumz(k)*xopt + matprod(xpt, w1*zmat(:, k)) !--!
128         vlag = Ax_plus_y(xpt, w1*zmat(:, k), qxoptq*sumz(k)*xopt)
129     !----------------------------------------------------------------------!
130         call r1update(bmat(:, 1:npt), ONE, vlag, zmat(:, k))
131         ! Implement R1UPDATE properly so that it ensures 
132         ! bmat(:, npt+1:npt+n) is symmetric.
133         call r1update(bmat(:, npt + 1 : npt + n), ONE, vlag)
134     end do
135     
136     !----------------------------------------------------------------!
137     ! The following instructions complete the shift of XBASE.
138     ! Recall the we have already subtracted HALF*XOPT from XPT. 
139     ! Therefore, overall, the new XPT is XPT - XOPT.
140     xpt = xpt - HALF*spread(xopt, dim = 2, ncopies = npt)
141     !do k = 1, npt
142     !    xpt(:, k) = xpt(:, k) - HALF*xopt
143     !end do
144     
145     xbase = xbase + xopt
146     xopt = ZERO
147     
148     end subroutine shiftbase
149     
150     
151     end module shiftbase_mod
