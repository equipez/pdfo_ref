2     !
3     ! Coded by Zaikun Zhang in July 2020 based on Powell's Fortran 77 code 
4     ! and the NEWUOA paper.
5     
6     
7     module newuob_mod
8     
9     implicit none
10     private
11     public :: newuob
12     
13     
14     contains
15     
16     
17     subroutine newuob(calfun, iprint, maxfun, npt, eta1, eta2, ftarget, gamma1, gamma2, rhobeg, rhoend, x, nf, f, fhist, xhist, info)
18     ! NEWUOB performs the actual calculations of NEWUOA. The arguments IPRINT,
19     ! MAXFUN, MAXHIST, NPT, ETA1, ETA2, FTARGET, GAMMA1, GAMMA2, RHOBEG, 
20     ! RHOEND, X, NF, F, FHIST, XHIST, and INFO are identical to the corresponding
21     ! arguments in subroutine NEWUOA. 
22     
23     ! XBASE will hold a shift of origin that should reduce the contributions 
24     ! from rounding errors to values of the model and Lagrange functions.
25     ! XOPT will be set to the displacement from XBASE of the vector of variables
26     ! that provides the least calculated F so far.
27     ! XNEW will be set to the displacement from XBASE of the vector of variables
28     ! for the current calculation of F.
29     ! XPT will contain the interpolation point coordinates relative to XBASE,
30     ! each COLUMN corresponding to a point.
31     ! FVAL will hold the values of F at the interpolation points.
32     ! GQ will hold the gradient of the quadratic model at XBASE.
33     ! HQ will hold the explicit second order derivatives of the quadratic model.
34     ! PQ will contain the parameters of the implicit second order derivatives
35     ! of the quadratic model.
36     ! BMAT will hold the last N ROWs of H. ZMAT will hold the factorization
37     ! of the leading NPT by NPT sub-matrix of H, this factorization being
38     ! ZMAT times Diag(DZ) times ZMAT^T, where DZ(1 : IDZ-1) = -1 and 
39     ! DZ(IDZ - 1 : NPT) = 1.
40     ! D is reserved for trial steps from XOPT.
41     ! VLAG will contain the values of the Lagrange functions at a new point
42     ! X. They are part of a product that requires VLAG to be of length NPT + N.
43     !
44     ! See Section 2 of the NEWUOA paper.
45     
46     ! Generic modules
47     use consts_mod, only : RP, IK, ZERO, HALF, TENTH, HUGENUM, DEBUGGING, SRNLEN
48     use info_mod, only : FTARGET_ACHIEVED, MAXFUN_REACHED, TRSUBP_FAILED, SMALL_TR_RADIUS, NAN_X, NAN_INF_F
49     use infnan_mod, only : is_nan, is_posinf
50     use debug_mod, only : errstop
51     use output_mod, only : retmssg, rhomssg, fmssg
52     use lina_mod, only : calquad, inprod
53     
54     ! Solver-specific modules
55     use pintrf_mod, only : FUNEVAL
56     use initialize_mod, only : initxf, initq, inith
57     use trustregion_mod, only : trsapp, trrad
58     use geometry_mod, only : setremove, ameliorgeo
59     use shiftbase_mod, only : shiftbase
60     use vlagbeta_mod, only : vlagbeta
61     use update_mod, only : updateh, updateq, tryqalt
62     
63     implicit none
64     
65     ! Inputs
66     procedure(FUNEVAL) :: calfun
67     integer(IK), intent(in) :: iprint
68     integer(IK), intent(in) :: maxfun
69     integer(IK), intent(in) :: npt
70     real(RP), intent(in) :: eta1
71     real(RP), intent(in) :: eta2
72     real(RP), intent(in) :: ftarget
73     real(RP), intent(in) :: gamma1
74     real(RP), intent(in) :: gamma2
75     real(RP), intent(in) :: rhobeg
76     real(RP), intent(in) :: rhoend
77     
78     ! In-output
79     real(RP), intent(inout) :: x(:)  ! SIZE(X) = N
80     
81     ! Outputs
82     integer(IK), intent(out) :: info
83     integer(IK), intent(out) :: nf
84     real(RP), intent(out) :: f
85     real(RP), intent(out) :: fhist(:)
86     real(RP), intent(out) :: xhist(:, :)
87     
88     ! Intermediate variables
89     integer(IK) :: idz 
90     integer(IK) :: ij(2, npt)
91     integer(IK) :: itest
92     integer(IK) :: khist
93     integer(IK) :: knew
94     integer(IK) :: kopt
95     integer(IK) :: maxfhist
96     integer(IK) :: maxtr
97     integer(IK) :: maxxhist
98     integer(IK) :: n
99     integer(IK) :: subinfo
100     integer(IK) :: tr
101     real(RP) :: beta
102     real(RP) :: bmat(size(x), npt + size(x))
103     real(RP) :: crvmin
104     real(RP) :: d(size(x))
105     real(RP) :: delbar
106     real(RP) :: delta
107     real(RP) :: distsq
108     real(RP) :: dnorm
109     real(RP) :: dnormsave(3)
110     real(RP) :: fopt
111     real(RP) :: moderr
112     real(RP) :: fsave
113     real(RP) :: fval(npt)
114     real(RP) :: gq(size(x))
115     real(RP) :: hq(size(x), size(x))
116     real(RP) :: moderrsave(size(dnormsave))
117     real(RP) :: pq(npt)
118     real(RP) :: ratio
119     real(RP) :: rho
120     real(RP) :: trtol
121     real(RP) :: vlag(npt + size(x))
122     real(RP) :: vquad
123     real(RP) :: xbase(size(x))
124     real(RP) :: xdsq(npt)
125     real(RP) :: xnew(size(x))
126     real(RP) :: xopt(size(x))
127     real(RP) :: xpt(size(x), npt)
128     real(RP) :: zmat(npt, npt - size(x) - 1)
129     logical :: improve_geometry
130     logical :: reduce_rho
131     logical :: shortd
132     logical :: terminate
133     character(len = 6), parameter :: solver = 'NEWUOA'
134     character(len = SRNLEN), parameter :: srname = 'NEWUOB'
135     
136     
137     ! Get size.
138     n = int(size(x), kind(n))
139     maxfhist = int(size(fhist), kind(maxfhist))
140     maxxhist = int(size(xhist, 2), kind(maxxhist))
141     
142     if (DEBUGGING) then
143         if (n == 0) then
144             call errstop(srname, 'SIZE(X) is invalid')
145         end if
146         if (size(xhist, 1) /= n .and. maxxhist > 0) then
147             call errstop(srname, 'XHIST is nonempty but SIZE(XHIST, 1) /= SIZE(X)')
148         end if
149         if (maxfhist*maxxhist > 0 .and. maxfhist /= maxxhist) then
150             call errstop(srname, 'FHIST and XHIST are both nonempty but SIZE(FHIST) /= SIZE(XHIST, 2)')
151         end if
152     end if
153     
154     maxtr = maxfun  ! Maximal number of trust region iterations.
155     terminate = .false. ! Whether to terminate after initialization.
156     
157     ! Initialize FVAL, XBASE, and XPT.
158     call initxf(calfun, iprint, x, rhobeg, ftarget, ij, kopt, nf, fhist, fval, xbase, xhist, xpt, subinfo)
159     xopt = xpt(:, kopt)
160     fopt = fval(kopt)
161     x = xbase + xopt  ! Set X.
162     f = fopt  ! Set F.
163     
164     ! Check whether to return.
165     if (subinfo == FTARGET_ACHIEVED) then
166         terminate = .true.
167     else if (subinfo == NAN_X) then
168         terminate = .true.
169     else if (subinfo == NAN_INF_F) then
170         terminate = .true.
171     end if
172     
173     if (terminate) then
174         info = subinfo
175         if (abs(iprint) >= 1) then
176             call retmssg(info, iprint, nf, f, x, solver)
177         end if
178         ! Rearrange FHIST and XHIST so that they are in the chronological order.
179         if (maxfhist >= 1 .and. maxfhist < nf) then
180             khist = mod(nf - 1_IK, maxfhist) + 1_IK
181             fhist = [fhist(khist + 1 : maxfhist), fhist(1 : khist)]
182         end if
183         if (maxxhist >= 1 .and. maxxhist < nf) then
184             khist = mod(nf - 1_IK, maxxhist) + 1_IK
185             xhist = reshape([xhist(:, khist + 1 : maxxhist), xhist(:, 1 : khist)], shape(xhist))
186             ! The above combination of SHAPE and RESHAPE fulfills our desire 
187             ! thanks to the COLUMN-MAJOR order of Fortran arrays.
188         end if
189         return
190     end if
191     
192     ! Initialize GQ, HQ, and PQ.
193     call initq(ij, fval, xpt, gq, hq, pq, subinfo)
194     
195     ! Initialize BMAT and ZMAT, and IDZ.
196     call inith(ij, xpt, idz, bmat, zmat, subinfo)
197     
198     ! After initializing GQ, HQ, PQ, BMAT, ZMAT, one can also choose to return
199     ! if subinfo = NAN_MODEL (NaN occurs in the model). We do not do it here.
200     ! If such a modle is harmful, then it will probably lead to other returns
201     ! (NaN in X, NaN in F, trust region subproblem fails, ...); otherwise, the
202     ! code will continue to run and possibly get rid of the NaN in the model.
203     
204     ! Set some more initial values.
205     rho = rhobeg
206     delta = rho
207     moderrsave = HUGENUM
208     dnormsave = HUGENUM
209     itest = 0
210     
211     ! Begin the iterative procedure.
212     do tr = 1, maxtr
213         ! Is the trust region trial step short?
214         shortd = .false.
215         ! Will we improve the model after the trust region iteration?
216         improve_geometry = .false.
217         ! Will we reduce rho after the trust region iteration?
218         reduce_rho = .false.
219         ! NEWUOA never sets IMPROVE_GEOMETRY and REDUCE_RHO to TRUE simultaneously.
220     
221         ! Solve the trust region subproblem.
222         ! In Powell's NEWUOA code, VQUAD is not an output of TRSAPP. Here we
223         ! output it but will NOT use it; it will sill be calculated latter by
224         ! CALQUAD in order to produce the same results as Powell's code.
225         trtol = 1.0e-2_RP  ! Tolerance used in trsapp.
226         call trsapp(delta, gq, hq, pq, trtol, xopt, xpt, crvmin, vquad, d, subinfo)
227     
228         ! Calculate the length of the trial step D.
229         dnorm = min(delta, sqrt(inprod(d, d)))
230     
231         ! Is the step long enough to invoke a function evaluation?
232         if (dnorm < HALF*rho) then
233             shortd = .true.
234             if (maxval(abs(moderrsave)) <= 0.125_RP*crvmin*rho*rho .and. maxval(dnormsave) <= rho) then
235                 ! Three recent values of ||D|| and |F−Q| are small.
236                 ! The 1st possibility (out of 2) that REDUCE_RHO = TRUE.
237                 reduce_rho = .true.
238             else 
239                 delta = TENTH*delta  ! Reduce DELTA by a factor of 10.
240                 if (delta <= 1.5_RP*rho) then
241                     delta = rho  ! Set DELTA to RHO when it is close.
242                 end if
243                 ! After this, DELTA < DNORM may happen, explaining why we 
244                 ! sometimes write MAX(DELTA, DNORM).
245             end if
246         end if
247     
248         if (.not. shortd) then  ! This is the normal case.
249             ! Save the current FOPT in FSAVE. It is needed later.
250             fsave = fopt
251     
252             ! Shift XBASE if XOPT may be too far from XBASE.
253             !if (inprod(d, d) <= 1.0e-3_RP*inprod(xopt, xopt)) then  ! Powell
254             if (dnorm*dnorm <= 1.0e-3_RP*inprod(xopt, xopt)) then
255                 call shiftbase(idz, pq, zmat, bmat, gq, hq, xbase, xopt, xpt)
256             end if
257     
258             ! Calculate VLAG and BETA for D.
259             call vlagbeta(idz, kopt, bmat, d, xopt, xpt, zmat, beta, vlag)
260     
261             ! Use the current quadratic model to predict the change in F due
262             ! to the step D.
263             call calquad(d, gq, hq, pq, xopt, xpt, vquad)
264     
265             ! Calculate the next value of the objective function.
266             xnew = xopt + d
267             x = xbase + xnew
268             if (any(is_nan(x))) then
269                 f = sum(x)  ! Set F to NaN. It is necessary.
270                 info = NAN_X
271                 exit
272             end if
273             call calfun(x, f)
274             nf = int(nf + 1, kind(nf))
275             if (abs(iprint) >= 3) then
276                 call fmssg(iprint, nf, f, x, solver)
277             end if
278             if (maxfhist >= 1) then
279                 khist = mod(nf - 1_IK, maxfhist) + 1_IK
280                 fhist(khist) = f
281             end if
282             if (maxxhist >= 1) then
283                 khist = mod(nf - 1_IK, maxxhist) + 1_IK
284                 xhist(:, khist) = x
285             end if
286     
287             ! FQDIFF is the error of the current model in predicting the change
288             ! in F due to D.
289             moderr = f - fsave - vquad
290     
291             ! Update FOPT and XOPT
292             if (f < fopt) then
293                 fopt = f
294                 xopt = xnew
295             end if
296     
297             ! Check whether to exit
298             if (is_nan(f) .or. is_posinf(f)) then
299                 info = NAN_INF_F
300                 exit
301             end if
302             if (f <= ftarget) then
303                 info = FTARGET_ACHIEVED
304                 exit
305             end if
306             if (nf >= maxfun) then
307                 info = MAXFUN_REACHED
308                 exit
309             end if
310     
311             ! Calculate the reduction ratio and update DELTA accordingly.
312             if (is_nan(vquad) .or. vquad >= ZERO) then
313                 info = TRSUBP_FAILED
314                 exit
315             end if
316             ratio = (f - fsave)/vquad
317             delta = trrad(delta, dnorm, eta1, eta2, gamma1, gamma2, ratio)
318             if (delta <= 1.5_RP*rho) then
319                 delta = rho
320             end if
321     
322             ! Set KNEW to the index of the interpolation point that will be
323             ! replaced by XNEW. KNEW will ensure that the geometry of XPT
324             ! is "optimal" after the replacement. Note that the information
325             ! of XNEW is included in VLAG and BETA, which are calculated 
326             ! according to D = XNEW - XOPT.
327             ! KNEW = 0 means it is not a good idea to replace any current 
328             ! interpolation point by XNEW.
329             call setremove(idz, kopt, beta, delta, ratio, rho, vlag(1:npt), xopt, xpt, zmat, knew)
330     
331             if (knew > 0) then  
332                 ! Update BMAT, ZMAT and IDZ, so that the KNEW-th interpolation
333                 ! point is replaced by XNEW.
334                 call updateh(knew, beta, vlag, idz, bmat, zmat)
335     
336                 ! Update the quadratic model using the updated BMAT, ZMAT, IDZ.
337                 call updateq(idz, knew, bmat(:, knew), moderr, zmat, xpt(:, knew), gq, hq, pq)
338     
339                 ! Include the new interpolation point. This should be done 
340                 ! after updating the model.
341                 fval(knew) = f
342                 xpt(:, knew) = xnew
343     
344                 ! Update KOPT to KNEW if F < FSAVE (i.e., last FOPT).
345                 if (f < fsave) then
346                     kopt = knew
347                 end if
348     
349                 ! Test whether to replace the new quadratic model Q by the least Frobenius 
350                 ! norm interpolant Q_alt. Perform the replacement if certain ceriteria is 
351                 ! satisfied. This part is OPTIONAL, but it is crucial for the performance on
352                 ! a certain class of problems. See Section 8 of the NEWUOA paper. 
353                 if (delta <= rho) then  ! DELTA == RHO.
354                     ! In theory, the FVAL - FSAVE in the following line can be replaced by 
355                     ! FVAL + C with any constant C. This constant will not affect the result 
356                     ! in precise arithmetic. Powell chose C = - FVAL(KOPT_ORIGINAL), where 
357                     ! KOPT_ORIGINAL is the KOPT before the update above (i.e., Powell updated
358                     ! KOPT after TRYQALT). Here we use the updated KOPT, because it worked 
359                     ! slightly better on CUTEst, although there is no difference theoretically.  
360                     ! Note that FVAL(KOPT_ORIGINAL) may not equal FSAVE --- it may happen that 
361                     ! KNEW = KOPT_ORIGINAL so that FVAL(KOPT_ORIGINAL) has been revised after 
362                     ! the last function evaluation.
363                     ! QUestion: Since TRYQALT is invoked only when DELTA equals the current RHO, 
364                     ! why not reset ITEST to 0 when RHO is reduced? 
365                     call tryqalt(idz, fval - fval(kopt), ratio, bmat(:, 1 : npt), zmat, itest, gq, hq, pq)
366                 end if
367             end if
368     
369             ! DNORMSAVE constains the DNORM corresponding to the latest 3
370             ! function evaluations with the current RHO.
371             dnormsave = [dnorm, dnormsave(1 : size(dnormsave) - 1)]
372             ! MODERR is the prediction errors of the latest 3 models.
373             moderrsave = [moderr, moderrsave(1 : size(moderrsave) - 1)]
374         end if
375     
376         ! The geometry of XPT probably needs improvement if
377         ! 1. The trust region step D is not short but RATIO < TENTH or there
378         ! is no good member in XPT to be replaced by XOPT + D (i.e., KNEW = 0), or
379         ! 2. D is short but the latest three model errors are not small enough 
380         ! to render REDUCE_RHO = TRUE.
381         if ((.not. shortd .and. (ratio < TENTH .or. knew == 0)) .or. (shortd .and. .not. reduce_rho)) then
382             ! Find out if the interpolation points are close enough to the 
383             ! best point so far, i.e., all the points are within a ball 
384             ! centered at XOPT with a radius of 2*DELTA. If not, set KNEW to
385             ! the index of the point that is the farthest away.
386             distsq = 4.0_RP*delta*delta
387             xdsq = sum((xpt - spread(xopt, dim = 2, ncopies = npt))**2, dim = 1)
388             if (maxval(xdsq) > distsq) then
389                 knew = int(maxloc(xdsq, dim = 1), kind(knew))
390                 distsq = maxval(xdsq)
391             else
392                 knew = 0
393             end if
394     
395             ! If KNEW is positive (i.e., not all points are close to XOPT),
396             ! then a model step will be taken to ameliorate the geometry of
397             ! the interpolation set and hence improve the model. Otherwise 
398             ! (all points are close to XOPT), RHO will be reduced 
399             ! (if MAX(DELTA, DNORM) <= RHO and D is "bad") or another 
400             ! trust-region step will be taken.
401             if (knew > 0) then
402                 ! The only possibility that IMPROVE_GEOMETRY = TRUE.
403                 improve_geometry = .true.
404             else if (max(delta, dnorm) <= rho .and. (ratio <= 0 .or. shortd)) then
405                 ! The 2nd possibility (out of 2) that REDUCE_RHO = TRUE.
406                 ! Even though all points are close to XOPT, a sufficiently
407                 ! small trust region does not suggest a good step to improve
408                 ! the current iterate. Then we should shrink RHO (i.e., update 
409                 ! the stadard for defining "closeness" and shortd).
410                 reduce_rho = .true.
411             end if
412         end if
413     
414         ! Before next trust region iteration, we may reduce rho or improve the
415         ! geometry of XPT according to REDUCE_RHO and IMPROVE_GEOMETRY.
416     
417         if (reduce_rho) then
418             ! The calculations with the current RHO are complete. Pick the 
419             ! next values of RHO and DELTA.
420             if (rho <= rhoend) then
421                 info = SMALL_TR_RADIUS  
422                 exit
423             else
424                 delta = HALF*rho
425                 ratio = rho/rhoend
426                 if (ratio <= 16.0_RP) then
427                     rho = rhoend
428                 else if (ratio <= 250.0_RP) then
429                     rho = sqrt(ratio)*rhoend
430                 else
431                     rho = TENTH*rho
432                 end if
433                 delta = max(delta, rho)
434                 ! DNORMSAVE constains the DNORM corresponding to the
435                 ! latest 3 function evaluations with the current RHO.
436                 dnormsave = HUGENUM
437                 moderrsave = HUGENUM
438                 if (abs(iprint) >= 2) then
439                     call rhomssg(iprint, nf, fopt, rho, xbase + xopt, solver)
440                end if
441             end if
442         end if  ! The procedure of reducing RHO ends.
443     
444         if (improve_geometry) then
445             ! Save the current FOPT in fsave. It is needed later.
446             fsave = fopt
447     
448             ! Set DELBAR, which will be used as the trust region radius for 
449             ! the geometry-improving schemes AMELIORGEO. We also need it to 
450             ! decide whether to shift XBASE or not.
451             delbar = max(min(TENTH*sqrt(distsq), HALF*delta), rho)
452     
453             ! Shift XBASE if XOPT may be too far from XBASE.
454             if (delbar*delbar<= 1.0e-3_RP*inprod(xopt, xopt)) then
455                 call shiftbase(idz, pq, zmat, bmat, gq, hq, xbase, xopt, xpt)
456             end if
457     
458             ! Find a step D so that the geometry of XPT will be improved 
459             ! when XPT(:, KNEW) is replaced by XOPT + D. The AMELIORGEO 
460             ! subroutine will call Powell's BIGLAG and BIGDEN. It will also
461             ! calculate the VLAG and BETA for this D.
462             call ameliorgeo(idz, knew, kopt, bmat, delbar, xopt, xpt, zmat, d, beta, vlag)
463     
464             ! Use the current quadratic model to predict the change in F due 
465             ! to the step D.
466             call calquad(d, gq, hq, pq, xopt, xpt, vquad)
467     
468             ! Calculate the next value of the objective function.
469             xnew = xopt + d
470             x = xbase + xnew
471             if (any(is_nan(x))) then
472                 f = sum(x)  ! Set F to NaN. It is necessary.
473                 info = NAN_X
474                 exit
475             end if
476             call calfun(x, f)
477             nf = int(nf + 1, kind(nf))
478             if (abs(iprint) >= 3) then
479                 call fmssg(iprint, nf, f, x, solver)
480             end if
481             if (maxfhist >= 1) then
482                 khist = mod(nf - 1_IK, maxfhist) + 1_IK
483                 fhist(khist) = f
484             end if
485             if (maxxhist >= 1) then
486                 khist = mod(nf - 1_IK, maxxhist) + 1_IK
487                 xhist(:, khist) = x
488             end if
489     
490             ! FQDIFF is the error of the current model in predicting the 
491             ! change in F due to D.
492             moderr = f - fsave - vquad
493     
494             ! Update FOPT and XOPT
495             if (f < fopt) then
496                 fopt = f
497                 xopt = xnew
498             end if
499     
500             ! Check whether to exit.
501             if (is_nan(f) .or. is_posinf(f)) then
502                 info = NAN_INF_F
503                 exit
504             end if
505             if (f <= ftarget) then
506                 info = FTARGET_ACHIEVED
507                 exit
508             end if
509             if (nf >= maxfun) then
510                 info = MAXFUN_REACHED
511                 exit
512             end if
513     
514             ! Update BMAT, ZMAT and IDZ, so that the KNEW-th interpolation 
515             ! point can be moved.
516             call updateh(knew, beta, vlag, idz, bmat, zmat)
517     
518             ! Update the quadratic model.
519             call updateq(idz, knew, bmat(:, knew), moderr, zmat, xpt(:, knew), gq, hq, pq)
520     
521             ! Include the new interpolation point. This should be done after
522             ! updating BMAT, ZMAT, and the model.
523             fval(knew) = f
524             xpt(:, knew) = xnew
525             if (f < fsave) then
526                 kopt = knew
527             end if
528     
529             ! DNORMSAVE constains the DNORM corresponding to the
530             ! latest 3 function evaluations with the current RHO.
531             !--------------------------------------------------------------!
532             ! Powell's code does not update DNORM. Therefore, DNORM is the 
533             ! length of last trust-region trial step, which seems inconsistent
534             ! with what is described in Section 7 (around (7.7)) of the NEWUOA
535             ! paper. Seemingly we should keep DNORM = ||D|| as we do here. The
536             ! value of DNORM will be used when defining REDUCE_RHO.
537             dnorm = min(delbar, sqrt(inprod(d, d)))
538             ! In theory, DNORM = DELBAR in this case.
539             !--------------------------------------------------------------!
540             dnormsave = [dnorm, dnormsave(1 : size(dnormsave) - 1)]
541             ! MODERR is the prediction errors of the latest 3 models.
542             moderrsave = [moderr, moderrsave(1 : size(moderrsave) - 1)]
543         end if  ! The procedure of improving geometry ends.
544     
545     end do
546     
547     ! Return from the calculation, after another Newton-Raphson step, if it
548     ! is too short to have been tried before.
549     ! Note that no trust region iteration has been done if MAXTR = 0, and
550     ! hence we should not check whether SHORTD = TRUE but return immediately.
551     if (maxtr > 0 .and. shortd .and. nf < maxfun) then
552         x = xbase + (xopt + d)
553         if (any(is_nan(x))) then
554             f = sum(x)  ! Set F to NaN. It is necessary.
555             info = NAN_X
556         else
557             call calfun(x, f)
558             nf = int(nf + 1, kind(nf))
559             if (abs(iprint) >= 3) then
560                 call fmssg(iprint, nf, f, x, solver)
561             end if
562             if (maxfhist >= 1) then
563                 khist = mod(nf - 1_IK, maxfhist) + 1_IK
564                 fhist(khist) = f
565             end if
566             if (maxxhist >= 1) then
567                 khist = mod(nf - 1_IK, maxxhist) + 1_IK
568                 xhist(:, khist) = x
569             end if
570         end if
571     end if
572     
573     ! Note that (FOPT .LE. F) is FALSE if F is NaN; When F is NaN, it is also
574     ! necessary to update X and F.
575     if (is_nan(f) .or. fopt <= f) then
576         x = xbase + xopt
577         f = fopt
578     end if
579     
580     ! Rearrange FHIST and XHIST so that they are in the chronological order.
581     if (maxfhist >= 1 .and. maxfhist < nf) then
582         khist = mod(nf - 1_IK, maxfhist) + 1_IK
583         fhist = [fhist(khist + 1 : maxfhist), fhist(1 : khist)]
584     end if
585     if (maxxhist >= 1 .and. maxxhist < nf) then
586         khist = mod(nf - 1_IK, maxxhist) + 1_IK
587         xhist = reshape([xhist(:, khist + 1 : maxxhist), xhist(:, 1 : khist)], shape(xhist))
588         ! The above combination of SHAPE and RESHAPE fulfills our desire 
589         ! thanks to the COLUMN-MAJOR order of Fortran arrays.
590     end if
591     
592     if (abs(iprint) >= 1) then
593         call retmssg(info, iprint, nf, f, x, solver)
594     end if
595     
596     return
597     end subroutine newuob
