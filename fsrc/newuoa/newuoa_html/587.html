27     subroutine alloc_rvector(x, n)
28     ! ALLOC_RVECTOR allocates the space for an allocatable REAL(RP)
29     ! vector X, whose size is N after allocation. 
30     use consts_mod, only : RP, IK, SRNLEN
31     use debug_mod, only : errstop
32     implicit none
33     
34     ! Input
35     integer(IK), intent(in) :: n
36     
37     ! Output
38     real(RP), allocatable, intent(out) :: x(:)
39     
40     ! Intermediate variable
41     integer :: alloc_status
42     character(len = SRNLEN), parameter :: srname = 'ALLOC_RVECTOR'
43     
44     ! According to the Fortran 2003 standard, when a procedure is invoked, 
45     ! any allocated ALLOCATABLE object that is an actual argument associated
46     ! with an INTENT(OUT) ALLOCATABLE dummy argument is deallocated. So it is
47     ! unnecessary to write the following line in F2003 since X is INTENT(OUT):
48     !!if (allocated(x)) deallocate (x)
49     
50     ! Allocate memory for X
51     allocate (x(n), stat = alloc_status)
52     if (alloc_status /= 0) then
53         call errstop(srname, 'Memory allocation fails.') 
54     end if
55     
56     ! Use X; otherwise, compilers may complain.
57     if (n >= 1) then
58         x(1) = 0.0_RP  
59     end if
60     
61     end subroutine alloc_rvector
62

64     subroutine alloc_rmatrix(x, m, n)
65     ! ALLOC_RMATRIX allocates the space for a REAL(RP) matrix X, whose
66     ! size is (M, N) after allocation.
67     use consts_mod, only : RP, IK, SRNLEN
68     use debug_mod, only : errstop
69     implicit none
70     
71     ! Input
72     integer(IK), intent(in) :: m, n
73     
74     ! Output
75     real(RP), allocatable, intent(out) :: x(:, :)
76     
77     ! Intermediate variable
78     integer :: alloc_status
79     character(len = SRNLEN), parameter :: srname = 'ALLOC_RMATRIX'
80     
81     ! Unnecessary to write the following line in F2003 since X is INTENT(OUT):
82     !!if (allocated(x)) deallocate (x)
83     
84     ! Allocate memory for X
85     allocate (x(m, n), stat = alloc_status)
86     if (alloc_status /= 0) then
87         call errstop(srname, 'Memory allocation fails.') 
88     end if
89     
90     ! Use X; otherwise, compilers may complain.
91     if (m*n >= 1) then
92         x(1, 1) = 0.0_RP
93     end if
94     
95     end subroutine alloc_rmatrix

98     pure function size_of_sp(x) result(y)
99     use consts_mod, only : SP, IK
100     implicit none
101     real(SP), intent(in) :: x
102     integer(IK) :: y
103     #if __USE_STORAGE_SIZE__ == 1
104         y = int(storage_size(x)/8, kind(y))
105     #else
106         y = int(kind(x), kind(y)) ! Avoid complaint 
107         y = int(4, kind(y))
108     #endif
109     end function size_of_sp
