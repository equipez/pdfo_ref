18     subroutine setremove(idz, kopt, beta, delta, ratio, rho, vlag, xopt, xpt, zmat, knew)
19     ! SETREMOVE sets KNEW to the index of the interpolation point that will 
20     ! be deleted AFTER A TRUST REGION STEP. KNEW will be set in a way
21     ! ensuring that the geometry of XPT is "optimal" after XPT(:, KNEW) is
22     ! replaced by XNEW = XOPT + D, where D is the trust-region step.  Note
23     ! that the information of XNEW is included in VLAG and BETA, which are
24     ! calculated according to D.
25     
26     ! Generic modules
27     use consts_mod, only : RP, IK, ONE, ZERO, TENTH, SRNLEN, DEBUGGING
28     use debug_mod, only : errstop, verisize
29     
30     implicit none
31     
32     ! Inputs
33     integer(IK), intent(in) :: idz
34     integer(IK), intent(in) :: kopt
35     real(RP), intent(in) :: beta
36     real(RP), intent(in) :: delta 
37     real(RP), intent(in) :: ratio 
38     real(RP), intent(in) :: rho 
39     real(RP), intent(in) :: vlag(:)  ! VLAG(NPT)
40     real(RP), intent(in) :: xopt(:)  ! XOPT(N)
41     real(RP), intent(in) :: xpt(:, :)  ! XPT(N, NPT)
42     real(RP), intent(in) :: zmat(:, :)  ! ZMAT(NPT, NPT - N - 1)
43     
44     ! Output
45     integer(IK), intent(out) :: knew
46     
47     ! Intermediate variables
48     integer(IK) :: n
49     integer(IK) :: npt
50     real(RP) :: hdiag(size(zmat, 1))
51     real(RP) :: rhosq 
52     real(RP) :: sigma(size(xpt, 2))
53     real(RP) :: xdsq(size(xpt, 2))
54     character(len = SRNLEN), parameter :: srname = 'SETREMOVE'
55          
56     
57     ! Get and verify the sizes
58     n = int(size(xpt, 1), kind(n))
59     npt = int(size(xpt, 2), kind(npt))
60     
61     if (DEBUGGING) then
62         if (n == 0 .or. npt < n + 2) then
63             call errstop(srname, 'SIZE(XPT) is invalid')
64         end if
65         call verisize(vlag, npt)
66         call verisize(xopt, n)
67         call verisize(zmat, npt, int(npt - n - 1, kind(n)))
68     end if
69     
70     rhosq = max(TENTH*delta, rho)**2
71     hdiag = -sum(zmat(:, 1 : idz - 1)**2, dim = 2) + sum(zmat(:, idz : npt - n - 1)**2, dim = 2)       
72     xdsq = sum((xpt - spread(xopt, dim = 2, ncopies = npt))**2, dim = 1)
73     sigma = abs(beta*hdiag + vlag(1 : npt)**2)
74     sigma = sigma * max(xdsq/rhosq, ONE)**3
75     if (ratio <= ZERO) then
76     ! When the new F is not better than the current FOPT,
77     ! we set SIGMA(KOPT) = -1 to prevent KNEW from being KOPT.
78         sigma(kopt) = -ONE
79     end if
80     if (maxval(sigma) > ONE .or. ratio > ZERO) then
81     ! KNEW > 0 unless MAXVAL(SIGMA) <= 1 and RATIO <= ZERO.
82     ! If RATIO > ZERO (i.e., the new F is smaller than the current FOPT),
83     ! then KNEW > 0, ensuring XNEW to be included into XPT.
84         knew = int(maxloc(sigma, dim = 1), kind(knew))
85     else
86         knew = 0
87     end if
88     end subroutine setremove

91     subroutine ameliorgeo(idz, knew, kopt, bmat, delbar, xopt, xpt, zmat, d, beta, vlag)
92     
93     ! Generic modules
94     use consts_mod, only : RP, IK, ONE, DEBUGGING, SRNLEN
95     use debug_mod, only : errstop, verisize
96     use lina_mod, only : inprod
97     
98     ! Solver-spcific modules
99     use vlagbeta_mod, only : vlagbeta
100     
101     implicit none
102     
103     ! Inputs
104     integer(IK), intent(in) :: idz
105     integer(IK), intent(in) :: knew
106     integer(IK), intent(in) :: kopt
107     real(RP), intent(in) :: bmat(:, :)  ! BMAT(N, NPT+N)
108     real(RP), intent(in) :: delbar
109     real(RP), intent(in) :: xopt(:)     ! XOPT(N)
110     real(RP), intent(in) :: xpt(:, :)   ! XPT(N, NPT)
111     real(RP), intent(in) :: zmat(:, :)  ! ZMAT(NPT, NPT - N - 1)
112     
113     ! In-output
114     real(RP), intent(inout) :: d(:)     ! D(N)
115     
116     ! Outputs
117     real(RP), intent(out) :: beta
118     real(RP), intent(out) :: vlag(:)    ! VLAG(NPT + N)
119     
120     ! Intermediate variables
121     integer(IK) :: n
122     integer(IK) :: npt
123     real(RP) :: alpha
124     real(RP) :: zknew(size(zmat, 2))
125     character(len = SRNLEN), parameter :: srname = 'AMELIORGEO'
126     
127     
128     ! Get and verify the sizes.
129     n = int(size(xpt, 1), kind(n))
130     npt = int(size(xpt, 2), kind(npt))
131     
132     if (DEBUGGING) then
133         if (n == 0 .or. npt < n + 2) then
134             call errstop(srname, 'SIZE(XPT) is invalid')
135         end if
136         call verisize(xopt, n)
137         call verisize(bmat, n, npt + n)
138         call verisize(zmat, npt, int(npt - n - 1, kind(n)))
139         call verisize(vlag, npt + n)
140         call verisize(d, n)
141     end if    
142     
143     call biglag(idz, knew, delbar, bmat, xopt, xpt, zmat, d)
144     
145     ! ALPHA is the KNEW-th diagonal entry of H
146     zknew = zmat(knew, :)
147     zknew(1 : idz - 1) = -zknew(1 : idz - 1)
148     alpha = inprod(zmat(knew, :), zknew)
149     
150     ! Calculate VLAG and BETA for D.
151     call vlagbeta(idz, kopt, bmat, d, xopt, xpt, zmat, beta, vlag)
152     
153     ! If the cancellation in DENOM is unacceptable, then BIGDEN calculates
154     ! an alternative model step D.  VLAG and BETA for this D are calculated
155     ! within BIGDEN. 
156     if (abs(ONE + alpha*beta/vlag(knew)**2) <= 0.8_RP) then
157         call bigden(idz, knew, kopt, bmat, xopt, xpt, zmat, d, beta, vlag)
158     end if
159     
160     end subroutine ameliorgeo

163     subroutine biglag(idz, knew, delbar, bmat, x, xpt, zmat, d)
164     ! BIGLAG calculates a D by approximately solving
165     !
166     ! max |LFUNC(X + D)|, subject to ||D|| <= DELBAR, 
167     !
168     ! where LFUNC is the KNEW-th Lagrange function.
169     ! See Setion 6 of the NEWUOA paper.
170     
171     ! Generic modules
172     use consts_mod, only : RP, IK, ONE, TWO, HALF, PI, ZERO, DEBUGGING, SRNLEN
173     use debug_mod, only : errstop, verisize
174     use lina_mod, only : Ax_plus_y, inprod, matprod
175     
176     implicit none
177     
178     ! Inputs
179     integer(IK), intent(in) :: idz
180     integer(IK), intent(in) :: knew
181     real(RP), intent(in) :: delbar 
182     real(RP), intent(in) :: bmat(:, :)  ! BMAT(N, NPT + N)
183     real(RP), intent(in) :: x(:)        ! X(N)
184     real(RP), intent(in) :: xpt(:, :)   ! XPT(N, NPT)
185     real(RP), intent(in) :: zmat(:, :)  ! ZMAT(NPT, NPT - N - 1)
186     
187     ! Output
188     real(RP), intent(out) :: d(:)       ! D(N)
189     
190     ! Intermediate variables
191     integer(IK) :: i
192     integer(IK) ::isave
193     integer(IK) ::iterc
194     integer(IK) ::iu
195     integer(IK) ::n
196     integer(IK) ::npt
197     real(RP) :: angle
198     real(RP) :: cf(5)
199     real(RP) :: cth
200     real(RP) :: dd
201     real(RP) :: denom
202     real(RP) :: dhd
203     real(RP) :: gc(size(x))
204     real(RP) :: gd(size(x))
205     real(RP) :: gg
206     real(RP) :: hcol(size(xpt, 2))
207     real(RP) :: s(size(x))
208     real(RP) :: scaling
209     real(RP) :: sp
210     real(RP) :: ss
211     real(RP) :: step
212     real(RP) :: sth
213     real(RP) :: t
214     real(RP) :: tau
215     real(RP) :: taua
216     real(RP) :: taub
217     real(RP) :: taubeg
218     real(RP) :: taumax
219     real(RP) :: tauold
220     real(RP) :: unitang
221     real(RP) :: w(size(x))
222     real(RP) :: zknew(size(zmat, 2))
223     character(len = SRNLEN), parameter :: srname = 'BIGLAG'
224     
225      
226     ! N is the number of variables.
227     ! NPT is the number of interpolation equations.
228     ! XPT contains the current interpolation points.
229     ! BMAT provides the last N ROWs of H.
230     ! ZMAT and IDZ give a factorization of the first NPT by NPT
231     ! sub-matrix of H.
232     ! KNEW is the index of the interpolation point to be removed.
233     ! DELBAR is the trust region bound.
234     ! D will be set to the step from X to the new point.
235     ! HCOL, GC, GD, S and W will be used for working space.
236     
237     
238     ! Get and verify the sizes.
239     n = int(size(xpt, 1), kind(n))
240     npt = int(size(xpt, 2), kind(npt))
241     
242     if (DEBUGGING) then
243         if (n == 0 .or. npt < n + 2) then
244             call errstop(srname, 'SIZE(XPT) is invalid')
245         end if
246         call verisize(x, n)
247         call verisize(bmat, n, npt + n)
248         call verisize(zmat, npt, int(npt - n - 1, kind(n)))
249     end if
250     
251     ! Set HCOL to the leading NPT elements of the KNEW-th column of H.
252     zknew = zmat(knew, :)
253     zknew(1 : idz - 1) = -zknew(1 : idz - 1)
254     hcol = matprod(zmat, zknew)
255     
256     ! Set the unscaled initial direction D. Form the gradient of LFUNC
257     ! at X, and multiply D by the Hessian of LFUNC.
258     d = xpt(:, knew) - x
259     dd = inprod(d, d)
260     
261     gd = matprod(xpt, hcol*matprod(d, xpt))
262     
263     !----------------------------------------------------------------!
264     !-----!gc = bmat(:, knew) + matprod(xpt, hcol*matprod(x, xpt)) !-!
265     gc = Ax_plus_y(xpt, hcol*matprod(x, xpt), bmat(:, knew))
266     !----------------------------------------------------------------!
267     
268     ! Scale D and GD, with a sign change if required. Set S to another
269     ! vector in the initial two dimensional subspace.
270     gg = inprod(gc, gc)
271     sp = inprod(d, gc)
272     dhd = inprod(d, gd)
273     scaling = delbar/sqrt(dd)
274     if (sp*dhd < ZERO) then 
275         scaling = - scaling
276     end if
277     t = ZERO
278     if (sp*sp > 0.99_RP*dd*gg) then 
279         t = ONE 
280     end if
281     tau = scaling*(abs(sp) + HALF*scaling*abs(dhd))
282     if (gg*(delbar*delbar) < 1.0e-2_RP*tau*tau) then 
283         t = ONE
284     end if
285     d = scaling*d
286     gd = scaling*gd
287     s = gc + t*gd
288     
289     ! Begin the iteration by overwriting S with a vector that has the
290     ! required length and direction, except that termination occurs if
291     ! the given D and S are nearly parallel.
292     do iterc = 1, n
293         dd = inprod(d, d)
294         sp = inprod(d, s)
295         ss = inprod(s, s)
296         if (dd*ss - sp*sp <= 1.0e-8_RP*dd*ss) then 
297             exit
298         end if
299         denom = sqrt(dd*ss - sp*sp)
300         s = (dd*s - sp*d)/denom
301     
302         w = matprod(xpt, hcol*matprod(s, xpt))
303         
304         ! Calculate the coefficients of the objective function on the
305         ! circle, beginning with the multiplication of S by the second
306         ! derivative matrix.
307         cf(1) = inprod(s, w)
308         cf(2) = inprod(d, gc)
309         cf(3) = inprod(s, gc)
310         cf(4) = inprod(d, gd)
311         cf(5) = inprod(s, gd)
312         cf(1) = HALF*cf(1)
313         cf(4) = HALF*cf(4) - cf(1)
314         
315         ! Seek the value of the angle that maximizes |TAU|.
316         taubeg = cf(1) + cf(2) + cf(4)
317         taumax = taubeg
318         tauold = taubeg
319         isave = 0
320         iu = 49
321         unitang = (TWO*PI)/real(iu + 1, RP)
322     
323         do i = 1, iu
324             angle = real(i, RP)*unitang
325             cth = cos(angle)
326             sth = sin(angle)
327             tau = cf(1) + (cf(2) + cf(4)*cth)*cth + (cf(3) + cf(5)*cth)*sth
328             if (abs(tau) > abs(taumax)) then
329                 taumax = tau
330                 isave = i
331                 taua = tauold
332             else if (i == isave + 1) then
333                 taub = tau
334             end if
335             tauold = tau
336         end do
337     
338         if (isave == 0) then 
339             taua = tau
340         end if
341         if (isave == iu) then 
342             taub = taubeg
343         end if
344         if (abs(taua - taub) > ZERO) then
345             taua = taua - taumax
346             taub = taub - taumax
347             step = HALF*(taua - taub)/(taua + taub)
348         else
349             step = ZERO
350         end if
351         angle = unitang*(real(isave, RP) + step)
352         
353         ! Calculate the new D and GD. Then test for convergence.
354         cth = cos(angle)
355         sth = sin(angle)
356         tau = cf(1) + (cf(2) + cf(4)*cth)*cth + (cf(3) + cf(5)*cth)*sth
357         d = cth*d + sth*s
358         gd = cth*gd + sth*w
359         s = gc + gd
360         if (abs(tau) <= 1.1_RP*abs(taubeg)) then
361             exit
362         end if
363     end do
364     
365     end subroutine biglag

368     subroutine bigden(idz, knew, kopt, bmat, x, xpt, zmat, d, beta, vlag)
369     ! BIGDEN calculates a D by approximately solving
370     !
371     ! max |SIGMA(X + D)|, subject to ||D|| <= DELBAR, 
372     !
373     ! where SIGMA is the denominator sigma in the updating formula
374     ! (4.11)--(4.12) for H, which is the inverse of the coefficient
375     ! matrix for the interplolation system (see (3.12)). Indeed, each
376     ! column of H corresponds to a Lagrange basis function of the
377     ! interpolation problem. 
378     ! In addition, it sets VLAG and BETA for the selected D.
379     ! See Setion 6 of the NEWUOA paper.
380     
381     ! Generic modules
382     use consts_mod, only : RP, IK, ONE, TWO, HALF, QUART, PI, ZERO, DEBUGGING, SRNLEN
383     use debug_mod, only : errstop, verisize
384     use lina_mod, only : Ax_plus_y, inprod, matprod
385     
386     implicit none
387     
388     ! Inputs
389     integer(IK), intent(in) :: idz
390     integer(IK), intent(in) :: knew
391     integer(IK), intent(in) :: kopt
392     real(RP), intent(in) :: bmat(:, :)  ! BMAT(N, NPT+N)
393     real(RP), intent(in) :: x(:)        ! X(N)
394     real(RP), intent(in) :: xpt(:, :)   ! XPT(N, NPT)
395     real(RP), intent(in) :: zmat(:, :)  ! ZMAT(NPT, NPT - N - 1)
396     
397     ! In-output
398     real(RP), intent(inout) :: d(:)     ! D(N)
399     
400     ! Outputs
401     real(RP), intent(out) :: beta
402     real(RP), intent(out) :: vlag(:)    ! VLAG(NPT + N)
403     
404     ! Intermediate variable
405     integer(IK) :: i
406     integer(IK) :: isave
407     integer(IK) :: iterc
408     integer(IK) :: iu
409     integer(IK) :: j
410     integer(IK) :: jc
411     integer(IK) :: k
412     integer(IK) :: n
413     integer(IK) :: npt
414     integer(IK) :: nw
415     real(RP) :: alpha
416     real(RP) :: angle
417     real(RP) :: dd
418     real(RP) :: den(9)
419     real(RP) :: dena
420     real(RP) :: denb
421     real(RP) :: denex(9)
422     real(RP) :: denmax
423     real(RP) :: denold
424     real(RP) :: denom
425     real(RP) :: denomold
426     real(RP) :: densav
427     real(RP) :: ds
428     real(RP) :: dstemp(size(xpt, 2))
429     real(RP) :: dtest
430     real(RP) :: dxn
431     real(RP) :: hcol(size(xpt, 2))
432     real(RP) :: par(9)       
433     real(RP) :: prod(size(xpt, 2) + size(x), 5)
434     real(RP) :: s(size(x)) 
435     real(RP) :: ss
436     real(RP) :: ssden
437     real(RP) :: sstemp(size(xpt, 2))
438     real(RP) :: step
439     real(RP) :: tau
440     real(RP) :: tempa
441     real(RP) :: tempb
442     real(RP) :: tempc
443     real(RP) :: unitang
444     real(RP) :: v(size(xpt, 2))
445     real(RP) :: w(size(xpt, 2) + size(x), 5)
446     real(RP) :: wz(size(zmat, 2))
447     real(RP) :: xd
448     real(RP) :: xnew(size(x))
449     real(RP) :: xnsq
450     real(RP) :: xptemp(size(xpt, 1), size(xpt, 2))
451     real(RP) :: xs
452     real(RP) :: xsq
453     real(RP) :: zknew(size(zmat, 2))
454     character(len = SRNLEN), parameter :: srname = 'BIGDEN'
455     
456     ! N is the number of variables.
457     ! NPT is the number of interpolation equations.
458     ! X is the best interpolation point so far.
459     ! XPT contains the current interpolation points.
460     ! BMAT provides the last N ROWs of H.
461     ! ZMAT and IDZ give a factorization of the first NPT by NPT
462     ! sub-matrix of H.
463     ! NDIM is the second dimension of BMAT and has the value NPT + N.
464     ! KOPT is the index of the optimal interpolation point.
465     ! KNEW is the index of the interpolation point to be removed.
466     ! D will be set to the step from X to the new point, and on 
467     ! entry it should be the D that was calculated by the last call
468     ! of BIGLAG. The length of the initial D provides a trust region
469     ! bound on the final D.
470     ! VLAG will be set to Theta*WCHECK+e_b for the final choice of D.
471     ! BETA will be set to the value that will occur in the updating
472     ! formula when the KNEW-th interpolation point is moved to its new
473     ! position.
474     
475     ! D is calculated in a way that should provide a denominator with 
476     ! a large modulus in the updating formula when the KNEW-th
477     ! interpolation point is shifted to the new position X + D.
478     
479      
480     ! Get and verify the sizes.
481     n = int(size(xpt, 1), kind(n))
482     npt = int(size(xpt, 2), kind(npt))
483     
484     if (DEBUGGING) then
485         if (n == 0 .or. npt < n + 2) then
486             call errstop(srname, 'SIZE(XPT) is invalid')
487         end if
488         call verisize(x, n)
489         call verisize(bmat, n, npt + n)
490         call verisize(zmat, npt, int(npt - n - 1, kind(n)))
491         call verisize(vlag, npt + n)
492         call verisize(d, n)
493     end if    
494     
495     ! Store the first NPT elements of the KNEW-th column of H in HCOL.
496     zknew = zmat(knew, :)
497     zknew(1 : idz - 1) = -zknew(1 : idz - 1)
498     hcol = matprod(zmat, zknew)
499     alpha = hcol(knew)
500     
501     ! The initial search direction D is taken from the last call of
502     ! BIGLAG, and the initial S is set below, usually to the direction
503     ! from X to X_KNEW, but a different direction to an 
504     ! interpolation point may be chosen, in order to prevent S from
505     ! being nearly parallel to D.
506     dd = inprod(d, d)
507     s = xpt(:, knew) - x
508     ds = inprod(d, s)
509     ss = inprod(s, s)
510     xsq = inprod(x, x)
511     
512     if (.not. (ds*ds <= 0.99_RP*dd*ss)) then
513         dtest = ds*ds/ss
514         xptemp = xpt - spread(x, dim = 2, ncopies = npt)
515     !----------------------------------------------------------------------!
516     !---------!dstemp = matprod(d, xpt) - inprod(x, d) !-------------------!
517         dstemp = matprod(d, xptemp)
518     !----------------------------------------------------------------------!
519         sstemp = sum((xptemp)**2, dim = 1) 
520         
521         dstemp(kopt) = TWO*ds + ONE  
522         sstemp(kopt) = ss     
523         k = int(minloc(dstemp*dstemp/sstemp, dim = 1), kind(k))
524         if ((.not. (dstemp(k)*dstemp(k)/sstemp(k) >= dtest)) .and. k /= kopt) then
525         ! Althoguh unlikely, if NaN occurs, it may happen that k=kopt.
526             s = xpt(:, k) - x
527             ds = dstemp(k)
528             ss = sstemp(k)
529         end if
530     end if
531     
532     ssden = dd*ss - ds*ds
533     densav = ZERO
534     
535     ! Begin the iteration by overwriting S with a vector that has the
536     ! required length and direction.
537     do iterc = 1, n
538         s = (ONE/sqrt(ssden))*(dd*s - ds*d)
539         xd = inprod(x, d)
540         xs = inprod(x, s)
541         
542         ! Set the coefficients of the first two terms of BETA.
543         tempa = HALF*xd*xd
544         tempb = HALF*xs*xs
545         den(1) = dd*(xsq + HALF*dd) + tempa + tempb
546         den(2) = TWO*xd*dd
547         den(3) = TWO*xs*dd
548         den(4) = tempa - tempb
549         den(5) = xd*xs
550         den(6 : 9) = ZERO
551         
552         ! Put the coefficients of WCHECK in W.
553         do k = 1, npt
554             tempa = inprod(xpt(:, k), d)
555             tempb = inprod(xpt(:, k), s)
556             tempc = inprod(xpt(:, k), x)
557             w(k, 1) = QUART*(tempa*tempa + tempb*tempb)
558             w(k, 2) = tempa*tempc
559             w(k, 3) = tempb*tempc
560             w(k, 4) = QUART*(tempa*tempa - tempb*tempb)
561             w(k, 5) = HALF*tempa*tempb
562         end do
563         w(npt + 1 : npt + n, 1 : 5) = ZERO
564         w(npt + 1 : npt + n, 2) = d
565         w(npt + 1 : npt + n, 3) = s
566         
567         ! Put the coefficents of THETA*WCHECK in PROD.
568         do jc = 1, 5
569             wz = matprod(w(1 : npt, jc), zmat)
570             wz(1 : idz - 1) = -wz(1 : idz - 1)
571             prod(1 : npt, jc) = matprod(zmat, wz)
572             
573             nw = npt
574             if (jc == 2 .or. jc == 3) then
575                 prod(1 : npt, jc) = prod(1 : npt, jc) + matprod(w(npt + 1 : npt + n, jc), bmat(:, 1 : npt))
576                 nw = npt + n
577             end if
578             prod(npt + 1 : npt + n, jc) = matprod(bmat(:, 1 : nw), w(1 : nw, jc))
579         end do
580         
581         ! Include in DEN the part of BETA that depends on THETA.
582         do k = 1, npt + n
583             par(1 : 5) = HALF*prod(k, 1 : 5)*w(k, 1 : 5)
584             den(1) = den(1) - par(1) - sum(par(1 : 5)) 
585             tempa = prod(k, 1)*w(k, 2) + prod(k, 2)*w(k, 1)
586             tempb = prod(k, 2)*w(k, 4) + prod(k, 4)*w(k, 2)
587             tempc = prod(k, 3)*w(k, 5) + prod(k, 5)*w(k, 3)
588             den(2) = den(2) - tempa - HALF*(tempb + tempc)
589             den(6) = den(6) - HALF*(tempb - tempc)
590             tempa = prod(k, 1)*w(k, 3) + prod(k, 3)*w(k, 1)
591             tempb = prod(k, 2)*w(k, 5) + prod(k, 5)*w(k, 2)
592             tempc = prod(k, 3)*w(k, 4) + prod(k, 4)*w(k, 3)
593             den(3) = den(3) - tempa - HALF*(tempb - tempc)
594             den(7) = den(7) - HALF*(tempb + tempc)
595             tempa = prod(k, 1)*w(k, 4) + prod(k, 4)*w(k, 1)
596             den(4) = den(4) - tempa - par(2) + par(3)
597             tempa = prod(k, 1)*w(k, 5) + prod(k, 5)*w(k, 1)
598             tempb = prod(k, 2)*w(k, 3) + prod(k, 3)*w(k, 2)
599             den(5) = den(5) - tempa - HALF*tempb
600             den(8) = den(8) - par(4) + par(5)
601             tempa = prod(k, 4)*w(k, 5) + prod(k, 5)*w(k, 4)
602             den(9) = den(9) - HALF*tempa
603         end do
604         
605         par(1 : 5) = HALF*prod(knew, 1 : 5)**2
606         denex(1) = alpha*den(1) + par(1) + sum(par(1 : 5))
607         tempa = TWO*prod(knew, 1)*prod(knew, 2)
608         tempb = prod(knew, 2)*prod(knew, 4)
609         tempc = prod(knew, 3)*prod(knew, 5)
610         denex(2) = alpha*den(2) + tempa + tempb + tempc
611         denex(6) = alpha*den(6) + tempb - tempc
612         tempa = TWO*prod(knew, 1)*prod(knew, 3)
613         tempb = prod(knew, 2)*prod(knew, 5)
614         tempc = prod(knew, 3)*prod(knew, 4)
615         denex(3) = alpha*den(3) + tempa + tempb - tempc
616         denex(7) = alpha*den(7) + tempb + tempc
617         tempa = TWO*prod(knew, 1)*prod(knew, 4)
618         denex(4) = alpha*den(4) + tempa + par(2) - par(3)
619         tempa = TWO*prod(knew, 1)*prod(knew, 5)
620         denex(5) = alpha*den(5) + tempa + prod(knew, 2)*prod(knew, 3)
621         denex(8) = alpha*den(8) + par(4) - par(5)
622         denex(9) = alpha*den(9) + prod(knew, 4)*prod(knew, 5)
623         
624         ! Seek the value of the angle that maximizes the |DENOM|.
625         denom = denex(1) + denex(2) + denex(4) + denex(6) + denex(8)
626         denold = denom
627         denmax = denom
628         isave = 0
629         iu = 49
630         unitang = (TWO*PI)/real(iu + 1, RP)
631         par(1) = ONE
632         do i = 1, iu
633             angle = real(i, RP)*unitang
634             par(2) = cos(angle)
635             par(3) = sin(angle)
636             do j = 4, 8, 2
637                 par(j) = par(2)*par(j - 2) - par(3)*par(j - 1)
638                 par(j + 1) = par(2)*par(j - 1) + par(3)*par(j - 2)
639             end do
640             denomold = denom
641             denom = inprod(denex(1 : 9), par(1 : 9))
642             if (abs(denom) > abs(denmax)) then
643                 denmax = denom
644                 isave = i
645                 dena = denomold
646             else if (i == isave + 1) then
647                 denb = denom
648             end if
649         end do
650         if (isave == 0) then
651             dena = denom
652         end if
653         if (isave == iu) then 
654             denb = denold
655         end if
656         if (abs(dena - denb) > 0) then
657             dena = dena - denmax
658             denb = denb - denmax
659             step = HALF*(dena - denb)/(dena + denb)
660         else
661             step = ZERO
662         end if
663         angle = unitang*(real(isave, RP) + step)
664         
665         ! Calculate the new parameters of the denominator, the new
666         ! VLAG vector and the new D. Then test for convergence.
667         par(2) = cos(angle)
668         par(3) = sin(angle)
669         do j = 4, 8, 2
670             par(j) = par(2)*par(j - 2) - par(3)*par(j - 1)
671             par(j + 1) = par(2)*par(j - 1) + par(3)*par(j - 2)
672         end do
673         
674         beta = inprod(den(1 : 9), par(1 : 9))
675         denmax = inprod(denex(1 : 9), par(1 : 9))
676         
677         vlag = matprod(prod(:, 1 : 5), par(1 : 5))
678         
679         tau = vlag(knew)
680         
681         d = par(2)*d + par(3)*s
682         dd = inprod(d, d)
683         xnew = x + d
684         dxn = inprod(d, xnew)
685         xnsq = inprod(xnew, xnew)
686         
687         if (iterc > 1) then
688             densav = max(densav, denold)
689         end if
690         if (abs(denmax) <= 1.1_RP*abs(densav)) then 
691             exit
692         end if
693         densav = denmax
694         
695         ! Set S to HALF the gradient of the denominator with respect 
696         ! to D. 
697         s = tau*bmat(:, knew) + alpha*(dxn*x + xnsq*d - vlag(npt + 1 : npt + n))
698         v = matprod(xnew, xpt)
699         v = (tau*hcol - alpha*vlag(1 : npt))*v
700     !----------------------------------------------------------------------!
701     !---------!s = s + matprod(xpt, v) !-----------------------------------!
702         s = Ax_plus_y(xpt, v, s)
703     !----------------------------------------------------------------------!
704         
705         ss = inprod(s, s)
706         ds = inprod(d, s)
707         ssden = dd*ss - ds*ds
708         if (ssden < 1.0e-8_RP*dd*ss) then 
709             exit
710         end if
711     end do
712     
713     vlag(kopt) = vlag(kopt) + ONE
714     
715     end subroutine bigden
