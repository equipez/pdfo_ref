17     subroutine trsapp(delta, gq, hq, pq, tol, x, xpt, crvmin, qred, s, info)
18     ! TRSAPP finds an approximate solution to the N-dimensional trust
19     ! region subproblem
20     !
21     ! min <X+S, GQ> + 0.5*<X+S, HESSIAN*(X+S)> s.t. ||S|| <= DELTA
22     !
23     ! Note that the HESSIAN here is the sum of an explicit part HQ and
24     ! an implicit part (PQ, XPT):
25     !
26     ! HESSIAN = HQ + sum_K=1^NPT PQ(K)*XPT(:, K)*XPT(:, K)' .
27     !
28     ! At return, S will be the approximate solution. CRVMIN will be 
29     ! set to the least curvature of HESSIAN along the conjugate 
30     ! directions that occur, except that it is set to ZERO if S goes
31     ! all the way to the trust region boundary. QRED is the reduction
32     ! of Q achieved by S. INFO is an exit flag:
33     ! INFO = 0: an approximate solution satisfying one of the 
34     ! following conditions is found: 
35     ! 1. ||G+HS||/||GBEG|| <= TOL, 
36     ! 2. ||S|| = DELTA and <S, -(G+HS)> >= (1 - TOL)*||S||*||G+HS||, 
37     ! where TOL is a tolerance that is set to 1e-2 in NEWUOA.  
38     ! INFO = 1: the iteration is reducing Q only slightly;
39     ! INFO = 2: the maximal number of iterations is attained;
40     ! INFO = -1: too much rounding error to continue
41     
42     ! The calculation of S begins with the truncated conjugate
43     ! gradient method. If the boundary of the trust region is reached, 
44     ! then further changes to S may be made, each one being in the 2D
45     ! space spanned by the current S and the corresponding gradient of
46     ! Q. Thus S should provide a substantial reduction to Q within the
47     ! trust region.
48     !
49     ! See Section 5 of the NEWUOA paper.
50     
51     ! Generic modules
52     use consts_mod, only : RP, IK, ONE, TWO, HALF, ZERO, PI, DEBUGGING, SRNLEN
53     use debug_mod, only : errstop, verisize
54     use infnan_mod, only : is_nan
55     use lina_mod, only : Ax_plus_y, inprod, matprod
56     
57     implicit none
58     
59     ! Inputs
60     real(RP), intent(in) :: delta 
61     real(RP), intent(in) :: gq(:)       ! GQ(N)
62     real(RP), intent(in) :: hq(:, :)    ! HQ(N, N)
63     real(RP), intent(in) :: pq(:)       ! PQ(NPT)
64     real(RP), intent(in) :: tol
65     real(RP), intent(in) :: x(:)        ! X(N)
66     real(RP), intent(in) :: xpt(:, :)   ! XPT(N, NPT)
67     
68     ! Outputs
69     integer(IK), intent(out) :: info
70     real(RP), intent(out) :: crvmin
71     real(RP), intent(out) :: qred
72     real(RP), intent(out) :: s(:)        ! S(N)
73     
74     ! Intermediate variables
75     integer(IK) :: i
76     integer(IK) :: isave
77     integer(IK) :: iterc
78     integer(IK) :: itermax
79     integer(IK) :: iu
80     integer(IK) :: n
81     integer(IK) :: npt
82     real(RP) :: alpha
83     real(RP) :: angle
84     real(RP) :: bstep
85     real(RP) :: cf
86     real(RP) :: cth
87     real(RP) :: d(size(x))
88     real(RP) :: dd
89     real(RP) :: delsq
90     real(RP) :: dg
91     real(RP) :: dhd
92     real(RP) :: dhs
93     real(RP) :: ds
94     real(RP) :: g(size(x))
95     real(RP) :: gg
96     real(RP) :: ggbeg
97     real(RP) :: ggsave
98     real(RP) :: hd(size(x))
99     real(RP) :: hs(size(x))
100     real(RP) :: hx(size(x)) 
101     real(RP) :: qadd
102     real(RP) :: qbeg
103     real(RP) :: qmin
104     real(RP) :: qnew
105     real(RP) :: qsave
106     real(RP) :: quada
107     real(RP) :: quadb
108     real(RP) :: reduc
109     real(RP) :: sg
110     real(RP) :: sgk
111     real(RP) :: shs
112     real(RP) :: ss
113     real(RP) :: sth
114     real(RP) :: t
115     real(RP) :: unitang
116     logical :: twod_search
117     character(len = SRNLEN), parameter :: srname = 'TRSAPP'
118     
119     
120     ! Get and verify the sizes.
121     n = int(size(xpt, 1), kind(n))
122     npt = int(size(xpt, 2), kind(npt))
123     
124     if (DEBUGGING) then
125         if (n == 0 .or. npt < n + 2) then
126             call errstop(srname, 'SIZE(XPT) is invalid')
127         end if
128         call verisize(gq, n)
129         call verisize(hq, n, n)
130         call verisize(pq, npt)
131         call verisize(s, n)
132     end if
133     
134     s = ZERO
135     crvmin = ZERO
136     qred = ZERO
137     info = 2  ! Default exit flag is 2, i.e., itermax is attained 
138     
139     ! Prepare for the first line search.
140     !----------------------------------------------------------------!
141     !-----!hx = matprod(xpt, pq*matprod(x, xpt)) + matprod(hq, x) !--!
142     hx = Ax_plus_y(hq, x, matprod(xpt, pq*matprod(x, xpt)))
143     !----------------------------------------------------------------!
144     g = gq + hx
145     gg = inprod(g, g)
146     ggbeg = gg
147     d = -g
148     dd = gg 
149     ds = ZERO
150     ss = ZERO
151     hs = ZERO
152     delsq = delta*delta
153     itermax = n
154     
155     twod_search = .false. 
156     
157     ! The truncated-CG iterations. 
158     !
159     ! The iteration will be terminated in 4 possible cases:
160     ! 1. the maximal number of iterations is attained;
161     ! 2. QADD <= TOL*QRED or ||G|| <= TOL*||GBEG||, where QADD is the
162     !    reduction of Q due to the latest CG step, QRED is the
163     !    reduction of Q since the begnning until the latest CG step, 
164     !    G is the current gradient, and GBEG is the initial gradient; 
165     !    see (5.13) of the NEWUOA paper;
166     ! 3. DS <= 0
167     ! 4. ||S|| = DELTA, i.e., CG path cuts the trust region boundary.
168     ! 
169     ! In the 4th case, twod_search will be set to true, meaning that S
170     ! will be improved by a sequence of two-dimensional search, the
171     ! two-dimensional subspace at each iteration being span(S, -G).
172     do iterc = 1, itermax
173         ! Check whether to exit due to small GG 
174         if (gg <= (tol**2)*ggbeg) then
175             info = 0
176             exit
177         end if
178         ! Set BSTEP to the step length such that ||S+BSTEP*D|| = DELTA
179         bstep = (delsq - ss)/(ds + sqrt(ds*ds + dd*(delsq - ss)))  
180     !----------------------------------------------------------------!
181     !-----!hd = matprod(xpt, pq*matprod(d, xpt)) + matprod(hq, d) !--------!
182     hd = Ax_plus_y(hq, d, matprod(xpt, pq*matprod(d, xpt)))
183     !----------------------------------------------------------------!
184         dhd = inprod(d, hd)
185     
186         ! Set the step-length ALPHA and update CRVMIN and 
187         if (dhd <= ZERO) then
188             alpha = bstep
189         else
190             alpha = min(bstep, gg/dhd)
191             if (iterc == 1) then
192                 crvmin = dhd/dd
193             else
194                 crvmin = min(crvmin, dhd/dd)
195             end if
196         end if
197         ! QADD is the reduction of Q due to the new CG step.
198         qadd = alpha*(gg - HALF*alpha*dhd)  
199         ! QRED is the reduction of Q up to now.
200         qred = qred + qadd
201         ! QADD and QRED will be used in the 2D minimization if any.
202     
203         ! Update S, HS, and GG.
204         s = s + alpha*d 
205         ss = inprod(s, s)
206         hs = hs + alpha*hd
207         ggsave = gg  ! Gradient norm square before this iteration 
208         gg = inprod(g + hs, g + hs)  ! Current gradient norm square
209         ! We may save g+hs for latter usage:
210         ! gnew = g + hs
211         ! Note that we should NOT set g = g + hs, because g contains
212         ! the gradient of Q at x.
213            
214         ! Check whether to exit. This should be done after updating HS
215         ! and GG, which will be used for the 2D minimization if any. 
216         if (alpha >= bstep .or. ss >= delsq) then
217             ! CG path cuts the boundary. Set CRVMIN to 0.
218             crvmin = ZERO
219             ! The only possibility that twod_search is true.
220             twod_search = .true.
221             exit
222         end if
223          
224         ! Check whether to exit due to small QADD
225         if (qadd <= tol*qred) then
226             info = 1 
227             exit
228         end if
229     
230         ! Prepare for the next CG iteration.
231         d = (gg/ggsave)*d - g - hs  ! CG direction
232         dd = inprod(d, d)
233         ds = inprod(d, s)
234         if (ds <= ZERO) then 
235             ! DS is positive in theory.
236             info = -1 
237             exit 
238         end if
239     end do 
240     
241     if (ss <= 0 .or. is_nan(ss)) then 
242         ! This may occur for ill-conditioned problems due to rounding.
243         info = -1
244         twod_search = .false.
245     end if
246     
247     if (twod_search) then
248         ! At least 1 iteration of 2D minimization
249         itermax = max(int(1, kind(itermax)), itermax - iterc)  
250     else
251         itermax = 0
252     end if
253     
254     ! The 2D minimization 
255     do iterc = 1, itermax
256         if (gg <= (tol**2)*ggbeg) then 
257             info = 0
258             exit
259         end if
260         sg = inprod(s, g)
261         shs = inprod(s, hs)
262         sgk = sg + shs
263         if (sgk/sqrt(gg*delsq) <= tol - ONE) then 
264             info = 0
265             exit
266         end if
267         
268         ! Begin the 2D minimization by calculating D and HD and some
269         ! scalar products.
270         t = sqrt(delsq*gg - sgk*sgk)
271         d = (delsq/t)*(g + hs) - (sgk/t)*s
272     !----------------------------------------------------------------!
273     !-----!hd = matprod(xpt, pq*matprod(d, xpt)) + matprod(hq, d) !--------!
274         hd = Ax_plus_y(hq, d, matprod(xpt, pq*matprod(d, xpt)))
275     !----------------------------------------------------------------!
276         dg = inprod(d, g)
277         dhd = inprod(hd, d)
278         dhs = inprod(hd, s)
279     
280         ! Seek the value of the angle that minimizes Q.
281         cf = HALF*(shs - dhd)
282         qbeg = sg + cf
283         qsave = qbeg
284         qmin = qbeg
285         isave = 0
286         iu = 49
287         unitang = (TWO*PI)/real(iu + 1, RP)
288     
289         do i = 1, iu
290             angle = real(i, RP)*unitang
291             cth = cos(angle)
292             sth = sin(angle)
293             qnew = (sg + cf*cth)*cth + (dg + dhs*cth)*sth
294             if (qnew < qmin) then
295                 qmin = qnew
296                 isave = i
297                 quada = qsave
298             else if (i == isave + 1) then
299                 quadb = qnew
300             end if
301             qsave = qnew
302         end do
303     
304         if (isave == 0) then
305             quada = qnew
306         end if
307         if (isave == iu) then 
308             quadb = qbeg
309         end if
310         if (abs(quada - quadb) > ZERO) then
311             quada = quada - qmin
312             quadb = quadb - qmin
313             angle = HALF*(quada - quadb)/(quada + quadb)
314         else
315             angle = ZERO
316         end if
317         angle = unitang*(real(isave, RP) + angle)
318     
319         ! Calculate the new S and HS. Then test for convergence.
320         cth = cos(angle)
321         sth = sin(angle)
322         reduc = qbeg - (sg + cf*cth)*cth - (dg + dhs*cth)*sth
323         s = cth*s + sth*d
324         hs = cth*hs + sth*hd
325         gg = inprod(g + hs, g + hs)
326         qred = qred + reduc
327         if (reduc/qred <= tol) then 
328             info = 1
329             exit
330         end if
331     end do
332     
333     end subroutine trsapp
334

336     function trrad(delta, dnorm, eta1, eta2, gamma1, gamma2, ratio)
337     
338     ! Generic module
339     use consts_mod, only : RP, HALF
340     
341     implicit none
342     
343     real(RP) :: trrad
344     real(RP), intent(in) :: delta  ! Current trust region radius
345     real(RP), intent(in) :: dnorm  ! Norm of current trust region step
346     real(RP), intent(in) :: eta1  ! Ratio threshold for contraction
347     real(RP), intent(in) :: eta2  ! Ratio threshold for expansion
348     real(RP), intent(in) :: gamma1 ! Contraction factor
349     real(RP), intent(in) :: gamma2 ! Expansion factor
350     real(RP), intent(in) :: ratio  ! Reduction ratio
351     
352     if (ratio <= eta1) then
353         trrad = gamma1*dnorm
354     else if (ratio <= eta2) then
355         trrad = max(HALF*delta, dnorm)
356     else
357         trrad = max(HALF*delta, gamma2*dnorm)
358     end if
359     
360     ! For noisy problems, the following may work better.
361     !if (ratio <= eta1) then
362     !trrad = gamma1*dnorm
363     !else if (ratio <= eta2) then  ! Ensure TRRAD >= DELTA
364     !trrad = delta  
365     !else  ! Ensure TRRAD > DELTA with a constant factor
366     !trrad = max(delta*(1.0_RP + gamma2)/2.0_RP, gamma2*dnorm)
367     !end if
368     
369     end function trrad
370     
371     
372     end module trustregion_mod
