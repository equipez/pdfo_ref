18     subroutine initxf(calfun, iprint, x, rhobeg, ftarget, ij, kopt, nf, fhist, fval, xbase, xhist, xpt, info)
19     ! INITXF performs the initialization regarding the interpolation 
20     ! points and corresponding function values. 
21     
22     ! Generic modules
23     use consts_mod, only : RP, IK, ZERO, DEBUGGING, SRNLEN
24     use debug_mod, only : errstop, verisize
25     use info_mod, only : FTARGET_ACHIEVED, NAN_X, NAN_INF_F
26     use infnan_mod, only : is_nan, is_posinf
27     use output_mod, only : fmssg
28     
29     ! Solver-specific module
30     use pintrf_mod, only : FUNEVAL
31     
32     implicit none
33     
34     ! Inputs
35     procedure(FUNEVAL) :: calfun
36     integer(IK), intent(in) :: iprint
37     real(RP), intent(in) :: x(:)  ! X(N)
38     real(RP), intent(in) :: rhobeg
39     real(RP), intent(in) :: ftarget
40     
41     ! Outputs
42     integer(IK), intent(out) :: info 
43     integer(IK), intent(out) :: ij(:, :)  ! IJ(2, NPT)
44     integer(IK), intent(out) :: kopt
45     integer(IK), intent(out) :: nf
46     real(RP), intent(out) :: fval(:)  ! FVAL(NPT)
47     real(RP), intent(out) :: fhist(:)  ! FHIST(MAXFHIST)
48     real(RP), intent(out) :: xbase(:)  ! XBASE(N)
49     real(RP), intent(out) :: xhist(:, :)  ! XHIST(MAXXHIST)
50     real(RP), intent(out) :: xpt(:, :)  ! XPT(N, NPT)
51     ! Remark on IJ:
52     ! When K > 2*N + 1, all the entries of XPT(:, K) will be zero except
53     ! that the IJ(1, K) and IJ(2, K) entries will be set to RHOBEG or 
54     ! -RHOBEG. Consequently, the Hessian of the quadratic model will get a
55     ! possibly nonzero (IJ(1, K), IJ(2, K)) entry. 
56     ! Indeed, IJ(:, 1 : 2*N + 1) is never used.
57     
58     ! Intermediate variables
59     integer(IK) :: i
60     integer(IK) :: itemp
61     integer(IK) :: j
62     integer(IK) :: k
63     integer(IK) :: khist
64     integer(IK) :: maxfhist
65     integer(IK) :: maxxhist
66     integer(IK) :: n
67     integer(IK) :: npt
68     integer(IK) :: npt_revised
69     real(RP) :: f
70     real(RP) :: xtemp(size(x))
71     logical :: evaluated(size(fval))
72     character(len = 6), parameter :: solver= 'NEWUOA'
73     character(len = SRNLEN), parameter :: srname = 'INITXF'
74     
75     
76     ! Get and verify the sizes.
77     n = int(size(x), kind(n))
78     npt = int(size(fval), kind(npt))
79     maxfhist = int(size(fhist), kind(maxfhist))
80     maxxhist = int(size(xhist, 2), kind(maxxhist))
81     
82     if (DEBUGGING) then
83         if (n == 0 .or. npt < n + 2) then
84             call errstop(srname, 'SIZE(X) or SIZE(FVAL) is invalid')
85         end if
86         if (size(xhist, 1) /= n .and. maxxhist > 0) then
87             call errstop(srname, 'XHIST is nonempty but SIZE(XHIST, 1) /= SIZE(X)')
88         end if
89         if (maxfhist*maxxhist > 0 .and. maxfhist /= maxxhist) then
90             call errstop(srname, 'FHIST and XHIST are both nonempty but SIZE(FHIST) /= SIZE(XHIST, 2)')
91         end if
92         call verisize(ij, 2_IK, npt)
93         call verisize(xbase, n)
94         call verisize(xpt, n, npt)
95     end if
96     
97     ! At return,
98     ! INFO = 0: initialization finishes normally
99     ! INFO = FTARGET_ACHIEVED: return because f <= ftarget
100     ! INFO = NAN_X: return because x contains NaN
101     ! INFO = NAN_INF_F: return because f is either NaN or +infinity
102     info = 0
103     
104     ! We set ij = 1 in case the initialization aborts due to abnormality. If
105     ! we do not do this, ij will be undefined if the initialization aborts.
106     ij = 1
107     
108     ! Set XBASE to X.
109     xbase = x
110     
111     ! Initialize XPT to ZERO.
112     xpt = ZERO
113     
114     ! Begin the initialization procedure. The coordinates of the
115     ! displacement of the next initial interpolation point from XBASE are
116     ! set in XPT(:, .).
117     
118     ! EVALUATED is a boolean array indicating whether the function value of
119     ! the i-th interpolation point has been evaluated. We need it for a
120     ! portable counting of the number of function evaluations, especially
121     ! if the loop is conducted asynchronously. However, the loop here is
122     ! not fully parallelizable if NPT>2N+1, because the definition
123     ! XPT(;, 2N+2:end) depends on FVAL(1:2N+1).
124     evaluated = .false.
125     
126     ! NPT_REVISED equals NPT, unless it turns out necessary to return due to
127     ! abnormality (NaN or Inf occurs, or F < FTARGET).
128     npt_revised = npt
129     
130     
131     ! Set XPT, FVAL, KOPT, FOPT, and XOPT.
132     
133     ! Set XPT(:, 2 : N + 1). 
134     do k = 2, min(npt, int(n + 1, kind(npt)))
135         xpt(k - 1, k) = rhobeg
136     end do
137     ! Set XPT(:, N+2 : NPT)
138     do k = int(n + 2, kind(k)), min(npt, int(2*n + 1, kind(npt)))
139         xpt(k - n - 1, k) = -rhobeg
140     end do
141      
142     ! Set FVAL(1 : NPT) by evaluating F. Totally parallelizable except for
143     ! FMSSG, which outputs messages to the console or files.
144     do k = 1, min(npt, int(2*n + 1, kind(npt)))
145         xtemp = xpt(:, k) + xbase
146         if (any(is_nan(xtemp))) then
147             f = sum(xtemp)  ! Set F to NaN. It is necessary.
148             info = NAN_X 
149             npt_revised = 0 
150             exit
151         end if
152         call calfun(xtemp, f)
153         evaluated(k) = .true.
154         fval(k) = f
155     
156         if (abs(iprint) >= 3) then
157             call fmssg(iprint, k, f, xtemp, solver) 
158         end if
159         if (maxfhist >= 1) then
160             khist = mod(k - 1_IK, maxfhist) + 1_IK
161             fhist(khist) = f
162         end if
163         if (maxxhist >= 1) then
164             khist = mod(k - 1_IK, maxxhist) + 1_IK
165             xhist(:, khist) = xtemp
166         end if
167     
168         ! Check whether to exit.
169         if (f <= ftarget) then
170             info = FTARGET_ACHIEVED 
171             npt_revised = 0 
172             exit
173         end if
174         if (is_posinf(f) .or. is_nan(f)) then
175             info = NAN_INF_F 
176             npt_revised = 0 
177             exit
178         end if
179     end do
180     
181     ! Set XPT(:, 2*N + 2 : NPT). It depends on FVAL(2 : 2*N + 1).
182     do k = int(2*n + 2, kind(k)), npt_revised
183         ! Decide IJ(:, K).  In general, when NPT = (N+1)*(N+2)/2, we can
184         ! set IJ(1 : NPT - (2*N + 1)) to ANY permutation of 
185         ! {(I, J) : 1 <= I > J <= N};
186         ! when NPT < (N+1)*(N+2)/2, IJ(1 : NPT - (2*N + 1)) is the 
187         ! first NPT - (2*N - 1) elements of such a permutation. Powell took
188         ! the following permutation:
189         itemp = int((k - n - 2)/n, kind(itemp))
190         j = int(k - (itemp + 1)*n - 1, kind(j))
191         i = j + itemp
192         if (i > n) then
193             itemp = j
194             j = i - n
195             i = itemp
196         end if
197         ij(1, k) = i
198         ij(2, k) = j
199     
200         ! The following lines set XPT(;, K) to 
201         ! XPT(:, I + 1) or XPT(:, I + N + 1) 
202         ! +
203         ! XPT(:, J + 1) or XPT(:, J + N + 1),
204         ! depending on the values of FVAL(I + 1), FVAL(I + N + 1),
205         ! FVAL(J + 1), and FVAL(J + N + 1). 
206         !
207         ! This is the only places where the definition
208         ! of XPT(:, 2*N + 2 : NPT) depends on F(2 : 2*N + 1).
209         ! If we set XPT(:, K) to XPT(:, I + 1) + XPT(:, J + 1)
210         ! regardless of FVAL, then the evaluations of FVAL(1 : NPT)
211         ! can be merged, and they are totally parallelizable; this can be
212         ! benificial if the function evaluations are expensive, which is
213         ! likely the case.
214         if (fval(i + 1) <= fval(i + n + 1)) then
215             xpt(i, k) = xpt(i, i + 1)
216         else
217             xpt(i, k) = xpt(i, i + n + 1)
218         end if
219         if (fval(j + 1) <= fval(j + n + 1)) then
220             xpt(j, k) = xpt(j, j + 1)
221         else
222             xpt(j, k) = xpt(j, j + n + 1)
223         end if
224     end do
225     
226     ! Set FVAL(2*N + 2 : NPT) by evaluating F. Totally parallelizable except
227     ! FMSSG, which outputs messages to the console or files.
228     do k = int(2*n + 2, kind(k)), npt_revised
229         xtemp = xpt(:, k) + xbase
230         if (any(is_nan(xtemp))) then
231             f = sum(xtemp)  ! Set F to NaN. It is necessary.
232             info = NAN_X
233             exit
234         end if
235         call calfun(xtemp, f)
236         evaluated(k) = .true.
237         fval(k) = f
238     
239         if (abs(iprint) >= 3) then
240             call fmssg(iprint, k, f, xtemp, solver) 
241         end if
242         if (maxfhist >= 1) then
243             khist = mod(k - 1_IK, maxfhist) + 1_IK
244             fhist(khist) = f
245         end if
246         if (maxxhist >= 1) then
247             khist = mod(k - 1_IK, maxxhist) + 1_IK
248             xhist(:, khist) = xtemp
249         end if
250         
251         ! Check whether to exit.
252         if (f <= ftarget) then
253             info = FTARGET_ACHIEVED 
254             exit
255         end if
256         if (is_posinf(f) .or. is_nan(f)) then
257             info = NAN_INF_F 
258             exit
259         end if
260     end do
261     
262     ! Set NF, KOPT
263     nf = int(count(evaluated), kind(nf))
264     kopt = int(minloc(fval, dim = 1, mask = evaluated), kind(kopt))
265     
266     end subroutine initxf

269     subroutine initq(ij, fval, xpt, gq, hq, pq, info)
270     ! INITQ initializes the quadratic model, which is represented by 
271     ! (GQ, HQ, PQ) in the following way:
272     ! the gradient of the model at XBASE is GQ;
273     ! the Hessian of the model is 
274     ! HQ + sum_{K=1}^NPT PQ(K)*XPT(:, K)*XPT(:, K)'.
275     
276     ! Generic modules
277     use consts_mod, only : RP, IK, ZERO, HALF, DEBUGGING, SRNLEN
278     use info_mod, only : NAN_MODEL
279     use debug_mod, only : errstop, verisize
280     use infnan_mod, only : is_nan
281     
282     implicit none
283     
284     ! Inputs
285     integer(IK), intent(in) :: ij(:, :)  ! IJ(2, NPT)
286     real(RP), intent(in) :: fval(:)  ! FVAL(NPT)
287     real(RP), intent(in) :: xpt(:, :)  ! XPT(N, NPT)
288     
289     ! Outputs
290     integer(IK), intent(out) :: info 
291     real(RP), intent(out) :: gq(:)  ! GQ(N)
292     real(RP), intent(out) :: hq(:, :)  ! HQ(N, N)
293     real(RP), intent(out) :: pq(:)  ! PQ(NPT)
294     
295     ! Intermediate variables
296     integer(IK) :: i
297     integer(IK) :: j
298     integer(IK) :: k
299     integer(IK) :: n
300     integer(IK) :: npt
301     real(RP) :: fbeg
302     real(RP) :: fi
303     real(RP) :: fj
304     real(RP) :: rhobeg
305     real(RP) :: xi
306     real(RP) :: xj
307     character(len = SRNLEN), parameter :: srname = 'INITQ'
308     
309     
310     ! Get and verify the sizes.
311     n = int(size(xpt, 1), kind(n))
312     npt = int(size(xpt, 2), kind(npt))
313     
314     if (DEBUGGING) then
315         if (n == 0 .or. npt < n + 2) then
316             call errstop(srname, 'SIZE(XPT) is invalid')
317         end if
318         call verisize(fval, npt)
319         call verisize(ij, 2_IK, npt)
320         call verisize(gq, n)
321         call verisize(hq, n, n)
322         call verisize(pq, npt)
323     end if
324     
325     gq = ZERO
326     hq = ZERO
327     pq = ZERO  ! We will not update PQ. It is ZERO at return.
328     
329     rhobeg = maxval(abs(xpt(:, 2)))  ! Read RHOBEG from XPT.
330     fbeg = fval(1)
331     
332     ! Set GQ by forward difference.
333     gq(1 : n) = (fval(2 : n + 1) - fbeg)/rhobeg
334     ! If possible, revise GQ to central difference. 
335     k = min(int(npt - n - 1, kind(n)), n)
336     gq(1 : k) = HALF*(gq(1 : k) + (fbeg - fval(n+2 : n+1+k))/rhobeg)
337     
338     ! Set the diagonal of HQ by 2nd-order central finite difference.
339     do k = 1, min(int(npt - n - 1, kind(n)), n) 
340         hq(k, k) = ((fval(k + 1) - fbeg)/rhobeg - (fbeg - fval(k + n + 1))/rhobeg)/rhobeg
341     end do
342     ! When NPT > 2*N + 1, set the off-diagonal entries of HQ.
343     do k = int(2*n + 2, kind(k)), npt 
344         ! I, J, XI, and XJ will be used below.
345         i = ij(1, k)
346         j = ij(2, k)
347         xi = xpt(i, k)
348         xj = xpt(j, k)
349         if (xi*xpt(i, i + 1) > ZERO) then 
350             fi = fval(i + 1)
351         else
352             fi = fval(i + n + 1)
353         end if
354         if (xj*xpt(j, j + 1) > ZERO) then
355             fj = fval(j + 1)
356         else
357             fj = fval(j + n + 1)
358         end if
359         ! With the XI, XJ, FI, and FJ found above, we have
360         ! FVAL(K) = F(XBASE + XI + XJ),
361         ! FI = F(XBASE + XI),
362         ! FJ = F(XBASE + XJ).
363         ! Thus the HQ(I, J) defined below approximates 
364         ! frac{partial^2}{partial X_I partial X_J} F(XBASE)
365         hq(i, j) = (fbeg - fi - fj + fval(k))/(xi*xj)
366         hq(j, i) = hq(i, j)
367     end do
368     
369     if (any(is_nan(gq)) .or. any(is_nan(hq)) .or. any(is_nan(pq)))then
370         info = NAN_MODEL 
371     else
372         info = 0
373     end if
374     
375     end subroutine initq

378     subroutine inith(ij, xpt, idz, bmat, zmat, info)
379     ! INITH initializes BMAT and ZMAT.
380     
381     ! Generic modules
382     use consts_mod, only : RP, IK, ZERO, ONE, HALF, DEBUGGING, SRNLEN
383     use info_mod, only : NAN_MODEL
384     use debug_mod, only : errstop, verisize
385     use infnan_mod, only : is_nan
386     
387     implicit none
388     
389     ! Inputs
390     integer(IK), intent(in) :: ij(:, :)  ! IJ(2, NPT)
391     real(RP), intent(in) :: xpt(:, :)  ! XPT(N, NPT)
392     
393     ! Outputs
394     integer(IK), intent(out) :: idz
395     integer(IK), intent(out) :: info
396     real(RP), intent(out) :: bmat(:, :)  ! BMAT(N, NPT + N)
397     real(RP), intent(out) :: zmat(:, :)  ! ZMAT(NPT, NPT - N - 1)
398     
399     ! Intermediate variables
400     integer(IK) :: i
401     integer(IK) :: j
402     integer(IK) :: k
403     integer(IK) :: n
404     integer(IK) :: npt
405     real(RP) :: recip
406     real(RP) :: reciq
407     real(RP) :: rhobeg
408     real(RP) :: rhosq
409     real(RP) :: xi 
410     real(RP) :: xj
411     character(len = SRNLEN), parameter :: srname = 'INITH'
412     
413     
414     ! Get and verify the sizes.
415     n = int(size(xpt, 1), kind(n))
416     npt = int(size(xpt, 2), kind(npt))
417     
418     if (DEBUGGING) then
419         if (n == 0 .or. npt < n + 2) then
420             call errstop(srname, 'SIZE(XPT) is invalid')
421         end if
422         call verisize(ij, 2_IK, npt)
423         call verisize(bmat, n, npt + n)
424         call verisize(zmat, npt, int(npt - n - 1, kind(n)))
425     end if
426     
427     ! Set IDZ = 1. It will not be changed in the following.
428     idz = 1
429     
430     ! Some values to be used for setting BMAT and ZMAT.
431     rhobeg = maxval(abs(xpt(:, 2)))  ! Read RHOBEG from XPT.
432     rhosq = rhobeg*rhobeg
433     recip = ONE/rhosq
434     reciq = sqrt(HALF)/rhosq
435     
436     ! Initialize BMAT and ZMAT to ZERO. 
437     bmat = ZERO
438     zmat = ZERO
439     
440     ! Set the nonzero initial elements of BMAT. 
441     ! When NPT >= 2*N + 1, this defines BMAT completely; 
442     ! When NPT <= 2*N, this defines BMAT(1 : NPT - N - 1, :).
443     do k = 1, min(int(npt - n - 1, kind(n)), n)
444         bmat(k, k + 1) = HALF/rhobeg
445         bmat(k, n + k + 1) = -HALF/rhobeg
446     end do
447     
448     ! When NPT <= 2*N, set BMAT(NPT - N : N, :)
449     do k = npt - n, n 
450         bmat(k, 1) = -ONE/rhobeg
451         bmat(k, k + 1) = ONE/rhobeg
452         bmat(k, npt + k) = -HALF*rhosq
453     end do
454               
455     ! Set the nonzero initial elements of ZMAT. 
456     ! When NPT <= 2*N + 1, this defines ZMAT completely; 
457     ! When NPT > 2*N + 1, this defines ZMAT(:, 1 : N).
458     do k = 1, min(int(npt - n - 1, kind(n)), n) 
459         zmat(1, k) = - reciq - reciq
460         zmat(k + 1, k) = reciq
461         zmat(k + n + 1, k) = reciq
462     end do
463     
464     ! When NPT > 2*N+1, set ZMAT(:, N + 1 : NPT - N - 1).
465     do k = int(n + 1, kind(k)), int(npt - n - 1, kind(k))
466         ! I, J, XI, and XJ will be used below.
467         i = ij(1, k + n + 1)
468         j = ij(2, k + n + 1)
469         xi = xpt(i, k + n + 1)
470         xj = xpt(j, k + n + 1)
471         if (xi < ZERO) then 
472             i = i + n
473         end if
474         if (xj < ZERO) then
475             j = j + n
476         end if
477         zmat(1, k) = recip
478         zmat(k + n + 1, k) = recip
479         zmat(i + 1, k) = -recip
480         zmat(j + 1, k) = -recip
481     end do
482     
483     if (any(is_nan(bmat)) .or. any(is_nan(zmat))) then
484         info = NAN_MODEL 
485     else
486         info = 0
487     end if
488     
489     end subroutine inith
490     
491     
492     end module initialize_mod
