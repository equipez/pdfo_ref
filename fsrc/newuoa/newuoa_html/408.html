1     ! PREPROC_MOD is a module that preprocess the inputs.
2     !
3     ! Coded by Zaikun Zhang in July 2020 based on Powell's Fortran 77 code 
4     ! and the NEWUOA paper.
5     
6     
7     module preproc_mod
8     
9     implicit none
10     private
11     public :: preproc
12     
13     
14     contains
15     
16     
17     subroutine preproc(n, iprint, maxfun, maxhist, npt, eta1, eta2, ftarget, gamma1, gamma2, rhobeg, rhoend) 
18     
19     use consts_mod, only : RP, IK, ZERO, ONE, TWO, HALF, TEN, TENTH, EPS
20     use consts_mod, only : RHOBEG_DFT, RHOEND_DFT, FTARGET_DFT, IPRINT_DFT
21     use infnan_mod, only : is_nan, is_inf, is_finite
22     
23     implicit none
24     
25     ! Input
26     integer(IK), intent(in) :: n
27     
28     ! In-outputs
29     integer(IK), intent(inout) :: iprint
30     integer(IK), intent(inout) :: maxfun
31     integer(IK), intent(inout) :: maxhist
32     integer(IK), intent(inout) :: npt
33     real(RP), intent(inout) :: eta1
34     real(RP), intent(inout) :: eta2
35     real(RP), intent(inout) :: ftarget
36     real(RP), intent(inout) :: gamma1
37     real(RP), intent(inout) :: gamma2
38     real(RP), intent(inout) :: rhobeg
39     real(RP), intent(inout) :: rhoend
40     
41     ! Intermediate variable
42     character(len = 6), parameter :: solver = 'NEWUOA'
43     
44     
45     if (iprint /= 0 .and. abs(iprint) /= 1 .and. abs(iprint) /= 2 .and. abs(iprint) /= 3) then
46         iprint = IPRINT_DFT
47         print '(/1A, I2, 1A)', solver // ': invalid IPRINT; it should be 0, 1, -1, 2, -2, 3, or -3; it is set to ', iprint, '.'
48     end if
49     
50     if (maxfun < n + 3) then
51         maxfun = int(n + 3, kind(maxfun))
52         print '(/1A, I8, 1A)', solver // ': invalid MAXFUN; it should an integer at least N + 3 ; it is set to ', maxfun, '.' 
53     end if
54     
55     if (maxhist < 0) then
56         maxhist = maxfun
57         print '(/1A, I8, 1A)', solver // ': invalid MAXHIST; it should be a nonnegative integer; it is set to ', maxhist, '.'
58     end if
59     ! MAXHIST > MAXFUN is never needed.
60     maxhist = min(maxhist, maxfun)
61     
62     if (npt < n + 2 .or. npt > min(maxfun - 1, ((n + 2)*(n + 1))/2)) then 
63         npt = int(min(maxfun - 1, 2*n + 1), kind(npt))
64         print '(/1A, I6, 1A)', solver // ': invalid NPT; it should an integer in the interval [N+2, (N+1)(N+2)/2], ' // &
65             & 'and it should be less than MAXFUN; it is set to ', npt, '.'
66     end if
67     
68     if (is_nan(ftarget)) then
69         ftarget = FTARGET_DFT
70         print '(/1A, 1PD15.6, 1A)', solver // ': invalid FTARGET; it should a real number; it is set to ', ftarget, '.'
71     end if
72     
73     ! When the difference between ETA1 and ETA2 is tiny, we force them to equal.
74     ! See the explanation around RHOBEG and RHOEND for the reason.
75     if (abs(eta1 - eta2) < 1.0e2_RP*EPS*max(abs(eta1), ONE)) then
76         eta2 = eta1
77     end if
78     
79     if (is_nan(eta1)) then
80         ! In this case, we take the value hard coded in Powell's orginal code
81         ! without any warning. It is useful when intefacing with MATLAB/Python.
82         eta1 = TENTH
83     else if (eta1 < ZERO .or. eta1 >= ONE) then
84         ! Take ETA1 into account if it has a valid value.
85         if (eta2 > ZERO .and. eta2 <= ONE) then
86             eta1 = max(EPS, eta2/7.0_RP) 
87         else
88             eta1 = TENTH
89         end if
90         print '(/1A, 1PD15.6, 1A)', solver // ': invalid ETA1; it should be in the interval [0, 1) and not more than ETA2;' // &
91             & ' it is set to ', eta1, '.' 
92     end if
93     
94     if (is_nan(eta2)) then
95         ! In this case, we take the value hard coded in Powell's orginal code
96         ! without any warning. It is useful when intefacing with MATLAB/Python.
97         eta2 = 0.7_RP
98     else if (eta2 < eta1 .or. eta2 > ONE) then
99         eta2 = (eta1 + TWO)/3.0_RP
100         print '(/1A, 1PD15.6, 1A)', solver // ': invalid ETA2; it should be in the interval [0, 1] and not less than ETA1;' // &
101             & ' it is set to ', eta2, '.'
102     end if
103     
104     if (is_nan(gamma1)) then
105         ! In this case, we take the value hard coded in Powell's orginal code
106         ! without any warning. It is useful when intefacing with MATLAB/Python.
107         gamma1 = HALF
108     else if (gamma1 <= ZERO .or. gamma1 >= ONE) then
109         gamma1 = HALF
110         print '(/1A, 1PD15.6, 1A)', solver // ': invalid GAMMA1; it should in the interval (0, 1); it is set to ', gamma1, '.'
111     end if
112     
113     if (is_nan(gamma2)) then
114         ! In this case, we take the value hard coded in Powell's orginal code
115         ! without any warning. It is useful when intefacing with MATLAB/Python.
116         gamma2 = TWO
117     else if (gamma2 < ONE .or. is_inf(gamma2)) then
118         gamma2 = TWO
119         print '(/1A, 1PD15.6, 1A)', solver // ': invalid GAMMA2; it should a real number not less than 1; it is set to ', &
120             & gamma2, '.' 
121     end if
122     
123     if (abs(rhobeg - rhoend) < 1.0e2_RP*EPS*max(abs(rhobeg), ONE))then
124     ! When the data is passed from the interfaces (e.g., MEX) to the Fortran
125     ! code, RHOBEG, and RHOEND may change a bit. It was oberved in a MATLAB
126     ! test that MEX passed 1 to Fortran as 0.99999999999999978. Therefore,
127     ! if we set RHOEND = RHOBEG in the interfaces, then it may happen that
128     ! RHOEND > RHOBEG, which is considered as an invalid input. To avoid this
129     ! situation, we force RHOBEG and RHOEND to equal when the difference is tiny.
130         rhoend = rhobeg
131     end if
132     
133     if (rhobeg <= ZERO .or. is_nan(rhobeg) .or. is_inf(rhobeg))then
134         ! Take RHOEND into account if it has a valid value.
135         if (is_finite(rhoend) .and. rhoend > ZERO) then
136             rhobeg = max(TEN*rhoend, RHOBEG_DFT)
137         else
138             rhobeg = RHOBEG_DFT
139         end if
140         print '(/1A, 1PD15.6, 1A)', solver // ': invalid RHOBEG; it should be a positive number; it is set to ', rhobeg, '.'
141     end if
142     
143     if (rhoend <= ZERO .or.  rhobeg < rhoend .or. is_nan(rhoend) .or. is_inf(rhoend)) then
144         rhoend = max(EPS, min(TENTH*rhobeg, RHOEND_DFT))
145         print '(/1A, 1PD15.6, 1A)', solver // ': invalid RHOEND; it should be a positive number and RHOEND <= RHOBEG; ' // &
146             & 'it is set to ', rhoend, '.'
147     end if
148     
149     end subroutine
150     
151     
152     end module preproc_mod
