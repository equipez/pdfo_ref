25     subroutine errstop(srname, mssg)
26     character(len = *), intent(in) :: srname
27     character(len = *), intent(in) :: mssg 
28     
29     #if __DEBUGGING__ == 1
30     call backtr()
31     #endif
32     
33     print '(/1A/)', 'Error: ' // trim(srname) // ': ' // trim(mssg) // '.'
34     
35     stop  ! This means to stop the whole program.
36     
37     end subroutine errstop

40     subroutine backtr()
41     ! BACKTR calls a compiler-dependent intrinsic to show a backtrace if we 
42     ! are in the debuge mode, i.e., __DEBUGGING__ == 1. 
43     ! N.B.: 
44     ! 1. The intrisic is compiler-dependent and does not exist in all 
45     ! compilers. Indeed, it is not standard-conforming. Therefore, compilers
46     ! may warn that a non-standard intrinsic is in use. 
47     ! 2. More seriously, if the compiler is instructed to conform to the 
48     ! standards (e.g., gfortran with the option -std=f2003) while __DEBUGGING__
49     ! is set to 1, then the compilation may FAIL at the linking stage, 
50     ! complaining that a subroutine cannot be found (e.g., backtrace() for
51     ! gfortran). In that case, we must set __DEBUGGING__ to 0 in ppf.h.
52     implicit none
53     #if __DEBUGGING__ == 1 
54     
55     #if defined __GFORTRAN__
56     call backtrace()
57     #elif defined __INTEL_COMPILER
58     call tracebackqq()
59     #endif
60     
61     #endif
62     end subroutine backtr

65     subroutine verisym(A, tol)
66     ! VERISYM verifies whether a matrix A is symmetric up to TOL.
67     use consts_mod, only : RP, ONE, ZERO, SRNLEN
68     implicit none
69     
70     real(RP), intent(in) :: A(:, :)
71     real(RP), intent(in) :: tol
72     
73     character(len = SRNLEN), parameter :: srname = 'VERISYM'
74     
75     if (size(A, 1) /= size(A, 2)) then
76         call errstop(srname, 'A is not square')
77     end if
78     if (tol > ZERO) then
79         if (maxval(abs((A - transpose(A)))) > tol*max(maxval(abs(A)), ONE)) then
80              call errstop(srname, 'A is not symmetric up to TOL')
81         end if
82     else
83         if (maxval(abs((A - transpose(A)))) > ZERO) then
84              call errstop(srname, 'A is not symmetric up to TOL')
85         end if
86     end if
87     end subroutine verisym

90     subroutine verisize_real_1(x, n)
91     ! VERISIZE_REAL_1 verifies whether SIZE(X) = N.
92     use consts_mod, only : RP, IK, SRNLEN
93     real(RP), intent(in) :: x(:)
94     integer(IK), intent(in) :: n
95     
96     character(len = SRNLEN), parameter :: srname = 'VERISIZE_REAL_1'
97     
98     if (size(x) /= n) then
99         call errstop(srname, 'SIZE(X) /= N')
100     end if
101     end subroutine verisize_real_1
